(*
	================================================================================
	Array sort template
	================================================================================

	This template generates typed Array_bubblesort, Array_mergesort, and
	Array_quicksort functions (each with an _n variant) from preprocessor macros.

	Usage:
	1. Define required macros.
	2. Include this template once per generated type.

	Required macros:
	  - ArraySortName
	    Suffix appended to function names.
	    Example: lreal  ->  Array_bubblesort_lreal, Array_quicksort_n_lreal, ...
	  - ArraySortElementType
	    Element type of the target array.
	    Example: lreal, dint

	Generated API:
	  - Array_bubblesort_n_@@ArraySortName@@(self, ascending, n)
	  - Array_bubblesort_@@ArraySortName@@(self, ascending)
	  - Array_mergesort_n_@@ArraySortName@@(self, ascending, n, work)
	  - Array_mergesort_@@ArraySortName@@(self, ascending, work)
	  - Array_quicksort_n_@@ArraySortName@@(self, ascending, n)
	  - Array_quicksort_@@ArraySortName@@(self, ascending)
	  - Convenience aliases:
	      Array_bubblesort_n -> Array_bubblesort_n_@@ArraySortName@@
	      Array_bubblesort   -> Array_bubblesort_@@ArraySortName@@
	      Array_mergesort_n  -> Array_mergesort_n_@@ArraySortName@@
	      Array_mergesort    -> Array_mergesort_@@ArraySortName@@
	      Array_quicksort_n  -> Array_quicksort_n_@@ArraySortName@@
	      Array_quicksort    -> Array_quicksort_@@ArraySortName@@

	Example (lreal):
	  {#define ArraySortName lreal}
	  {#define ArraySortElementType lreal}
	  {#include <array/sort_template.txt>}
	================================================================================
*)
{#include <sys.txt>}

{#ifndef ArraySortName}
	{#error require ArraySortName macro.}
{#endif}

{#ifndef ArraySortElementType}
	{#error require ArraySortElementType macro.}
{#endif}

{#define ARRAY_MERGESORT(T)       _ARRAY_MERGESORT_(T)}
{#define Array_mergesort          ARRAY_MERGESORT(ArraySortName)}

{#define ARRAY_QUICKSORT(T)       _ARRAY_QUICKSORT_(T)}
{#define Array_quicksort          ARRAY_QUICKSORT(ArraySortName)}

(*{doc Sort the first n elements of a 1-D array using bubble sort.
@inout self Array to sort.
@input ascending Sort ascending when true, descending when false.
@input n Number of elements to sort; -1 sorts the entire array.
@error If n exceeds the array length.}*)
function Array_bubblesort_n
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc Array to sort.}*)
		self: array[*] of ArraySortElementType;
	end
	var_input
		(*{doc Sort ascending when true, descending when false.}*)
		ascending: bool := true;
		(*{doc Number of elements to sort; -1 sorts the entire array.}*)
		n: dint := -1;
	end
	var_temp
		actual_n: dint;
		i_0, i_1: dint;
		i, j: dint;
		cond: bool;
		tmp: ArraySortElementType;
	end
	{st}
	i_0 := lower_bound(self, 1);
	i_1 := upper_bound(self, 1);
	if i_0 > i_1 then
		eno := false;
		return;
	end_if;
	actual_n := i_1 - i_0 + 1;
	if n >= 0 then
		if n > actual_n then
			eno := false;
			return;
		end_if;
		actual_n := n;
	end_if;

	for i := i_0 to i_0 + actual_n - 2 by 1 do
		j := i_0 + actual_n - 1;
		while j >= i + 1 do
			if ascending then
				cond := self[j - 1] > self[j];
			else
				cond := self[j - 1] < self[j];
			end_if;
			if cond then
				tmp := self[j - 1];
				self[j - 1] := self[j];
				self[j] := tmp;
			end_if;
			j := j - 1;
		end_while;
	end_for;
	{end}
end

function Array_bubblesort
	_PROLOGUE_OF_FUNCTION_()

(*{doc Sort the first n elements of a 1-D array using merge sort.
Requires a work array of at least n elements.
@inout self Array to sort.
@input ascending Sort ascending when true, descending when false.
@input n Number of elements to sort; -1 sorts the entire array.
@inout work Scratch array of at least n elements.
@error If n exceeds the array length or work is too small.}*)
function Array_mergesort_n
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc Array to sort.}*)
		self: array[*] of ArraySortElementType;
	end
	var_input
		(*{doc Sort ascending when true, descending when false.}*)
		ascending: bool := true;
		(*{doc Number of elements to sort; -1 sorts the entire array.}*)
		n: dint := -1;
	end
	var_in_out
		(*{doc Scratch array of at least n elements.}*)
		work: array[*] of ArraySortElementType;
	end
	var_temp
		i_0, i_1: dint;
		actual_n: dint;
		nself, nwork, width: dint;
		left_idx, mid_idx, right_idx: dint;
		l_idx, r_idx, dest: dint;
		w_i_0, w_i_1: dint;
		cond: bool;
	end
	{st}
	i_0 := lower_bound(self, 1);
	i_1 := upper_bound(self, 1);
	if i_0 > i_1 then
		eno := false;
		return;
	end_if;
	actual_n := i_1 - i_0 + 1;
	if n >= 0 then
		if n > actual_n then
			eno := false;
			return;
		end_if;
		actual_n := n;
	end_if;

	w_i_0 := lower_bound(work, 1);
	w_i_1 := upper_bound(work, 1);
	if actual_n > w_i_1 - w_i_0 + 1 then
		eno := false;
		return;
	end_if;

	if actual_n <= 1 then
		return;
	end_if;

	width := 1;
	while width < actual_n do
		left_idx := i_0;
		while left_idx <= i_0 + actual_n - 1 do
			mid_idx := left_idx + width - 1;
			if mid_idx >= i_0 + actual_n - 1 then
				exit;
			end_if;
			right_idx := left_idx + width + width - 1;
			if right_idx > i_0 + actual_n - 1 then
				right_idx := i_0 + actual_n - 1;
			end_if;
			l_idx := left_idx;
			r_idx := mid_idx + 1;
			dest := w_i_0 + (left_idx - i_0);
			while (l_idx <= mid_idx) and (r_idx <= right_idx) do
				if ascending then
					cond := self[l_idx] <= self[r_idx];
				else
					cond := self[l_idx] >= self[r_idx];
				end_if;
				if cond then
					work[dest] := self[l_idx];
					l_idx := l_idx + 1;
				else
					work[dest] := self[r_idx];
					r_idx := r_idx + 1;
				end_if;
				dest := dest + 1;
			end_while;
			while l_idx <= mid_idx do
				work[dest] := self[l_idx];
				dest := dest + 1;
				l_idx := l_idx + 1;
			end_while;
			while r_idx <= right_idx do
				work[dest] := self[r_idx];
				dest := dest + 1;
				r_idx := r_idx + 1;
			end_while;
			dest := w_i_0 + (left_idx - i_0);
			for l_idx := left_idx to right_idx do
				self[l_idx] := work[dest];
				dest := dest + 1;
			end_for;
			left_idx := right_idx + 1;
		end_while;
		width := width + width;
	end_while;
	{end}
end

(*{doc Sort a 1-D array using merge sort.
Requires a work array of at least as many elements as self.
@inout self Array to sort.
@input ascending Sort ascending when true, descending when false.
@inout work Scratch array of at least length(self) elements.}*)
function Array_mergesort
	_PROLOGUE_OF_FUNCTION_()

	var_in_out

	{end}
end

(*{doc Sort the first n elements of a 1-D array using quicksort.
@inout self Array to sort.
@input ascending Sort ascending when true, descending when false.
@input n Number of elements to sort; -1 sorts the entire array.
@error If n exceeds the array length.}*)
function Array_quicksort_n
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc Array to sort.}*)
		self: array[*] of ArraySortElementType;
	end
	var_input
		(*{doc Sort ascending when true, descending when false.}*)
		ascending: bool := true;
		(*{doc Number of elements to sort; -1 sorts the entire array.}*)
		n: dint := -1;
	end
	var_temp
		actual_n: dint;
		i_0, i_1: dint;
		left_idx, right_idx: dint;
		l_ptr, r_ptr: dint;
		pivot: ArraySortElementType;
		tmp: ArraySortElementType;
		stack_lo: array[0..63] of dint;
		stack_hi: array[0..63] of dint;
		sp: dint;
		cond: bool;
	end
	{st}
	i_0 := lower_bound(self, 1);
	i_1 := upper_bound(self, 1);
	if i_0 > i_1 then
		eno := false;
		return;
	end_if;
	actual_n := i_1 - i_0 + 1;
	if n >= 0 then
		if n > actual_n then
			eno := false;
			return;
		end_if;
		actual_n := n;
	end_if;

	sp := 0;
	stack_lo[0] := i_0;
	stack_hi[0] := i_0 + actual_n - 1;

	while sp >= 0 do
		left_idx := stack_lo[sp];
		right_idx := stack_hi[sp];
		sp := sp - 1;

		l_ptr := left_idx;
		r_ptr := right_idx;
		pivot := self[left_idx + (right_idx - left_idx) / 2];

		while l_ptr <= r_ptr do
			while TRUE do
				if ascending then
					cond := self[l_ptr] < pivot;
				else
					cond := self[l_ptr] > pivot;
				end_if;
				if cond then
					l_ptr := l_ptr + 1;
				else
					exit;
				end_if;
			end_while;

			while TRUE do
				if ascending then
					cond := self[r_ptr] > pivot;
				else
					cond := self[r_ptr] < pivot;
				end_if;
				if cond then
					r_ptr := r_ptr - 1;
				else
					exit;
				end_if;
			end_while;

			if l_ptr <= r_ptr then
				tmp := self[l_ptr];
				self[l_ptr] := self[r_ptr];
				self[r_ptr] := tmp;
				l_ptr := l_ptr + 1;
				r_ptr := r_ptr - 1;
			end_if;
		end_while;

		if left_idx < r_ptr then
			if l_ptr < i_0 + actual_n - 1 then
				if (r_ptr - left_idx) > (i_0 + actual_n - 1 - l_ptr) then
					if sp < 63 then
						sp := sp + 1;
						stack_lo[sp] := left_idx;
						stack_hi[sp] := r_ptr;
					end_if;
					if sp < 63 then
						sp := sp + 1;
						stack_lo[sp] := l_ptr;
						stack_hi[sp] := right_idx;
					end_if;
				else
					if sp < 63 then
						sp := sp + 1;
						stack_lo[sp] := l_ptr;
						stack_hi[sp] := right_idx;
					end_if;
					if sp < 63 then
						sp := sp + 1;
						stack_lo[sp] := left_idx;
						stack_hi[sp] := r_ptr;
					end_if;
				end_if;
			else
				if sp < 63 then
					sp := sp + 1;
					stack_lo[sp] := left_idx;
					stack_hi[sp] := r_ptr;
				end_if;
			end_if;
		elsif l_ptr < right_idx then
			if sp < 63 then
				sp := sp + 1;
				stack_lo[sp] := l_ptr;
				stack_hi[sp] := right_idx;
			end_if;
		end_if;
	end_while;
	{end}
end

@inout self Array to sort.
@input ascending Sort ascending when true, descending when false.}*)
function Array_quicksort
	_PROLOGUE_OF_FUNCTION_()

		self: array[*] of ArraySortElementType;
	end
	var_input

		ascending: bool := true;
	end
	{st}

	{end}
end
