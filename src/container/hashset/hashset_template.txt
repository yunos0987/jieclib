(*
	================================================================================
	HashSet Template Generator - Macro-based Generic HashSet
	================================================================================

	This template generates a typed HashSet from preprocessor macros.
	The internal implementation is based on HashMap.

	Usage:
	1. Define required macros.
	2. Optionally define behavior macros.
	3. Include this template once per generated type.

	Required macros:
	  - HashSetName
	    Suffix used for generated types and functions.
	  - HashSetElementType
	    Element type stored in the set.
	  - HashSetElementHashFunction
	    Element hash function.
	  - HashSetCapacity
	    Number of table slots.

	Optional macros:
	  - HashSetMaxLoadPermille
	    Maximum load factor in permille.
	    Default: 800 (80%).
	  - HashSetElementEqualsFunction
	    Element equality function.
	    Default: direct value comparison.

	Generated APIs:
	  - init, copy, add, add_all
	  - remove, remove_all, contains, len
	  - elements, equals, hash, empty, clear

	Example (string):
	  {#define HashSetName str}
	  {#define HashSetCapacity 256}
	  {#define HashSetElementType string[128]}
	  {#define HashSetElementHashFunction String_hash}
	  {#include <container/hashset/hashset_template.txt>}

	Example (dint):
	  {#define HashSetName dint}
	  {#define HashSetCapacity 256}
	  {#define HashSetElementType dint}
	  {#define HashSetElementHashFunction Dint_hash}
	  {#include <container/hashset/hashset_template.txt>}

	Algorithm:
	  HashMap-backed set with Robin Hood hashing.
	================================================================================
*)

(*{doc Add one element.
Insert the element if it is not already present.
@inout self Target HashSet.
@input element Element to add.
@output added true when newly added, false when already present.
@return true on success.
@error Propagates ENO from backing HashMap operations. }*)
function HashSet_add: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: HashSetCoreType;
	end
	var_input
		element: HashSetElementType;
	end
	var_output
		added: bool;
	end
	var_temp
		replaced: bool;
	end
	{st}
	HashSet_add := HashSet_backing_upsert(self:=self.backing, key:=element, value:=true, replaced=>replaced, eno=>eno);
	if not eno then
		return;
	end_if;
	added := not replaced;
	{end}
end

(*{doc Add all elements from an input array.
Process input array in index order and count new insertions.
@inout self Target HashSet.
@inout elements Input elements.
@output added_count Number of newly added elements.
@return true on success, false if an insert fails.
@error ENO is propagated from backing HashMap. }*)
function HashSet_add_all: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: HashSetCoreType;
	end
	var_in_out
		elements: array[*] of HashSetElementType;
	end
	var_output
		added_count: dint;
	end
	var_temp
		idx: dint;
		idx_begin, idx_end: dint;
		replaced: bool;
		succeeded: bool;
	end
	{st}
	idx_begin := lower_bound(elements, 1);
	idx_end := upper_bound(elements, 1);
	added_count := 0;

	for idx := idx_begin to idx_end by 1 do
		succeeded := HashSet_backing_upsert(self:=self.backing, key:=elements[idx], value:=true, replaced=>replaced, eno=>eno);
		if not eno then
			return;
		end_if;
		if not succeeded then
			HashSet_add_all := false;
			return;
		end_if;
		if not replaced then
			added_count := added_count + 1;
		end_if;
	end_for;

	HashSet_add_all := true;
	{end}
end

(*{doc Remove all matching elements from an input array.
Process input array in index order and count successful deletions.
@inout self Target HashSet.
@inout elements Input elements.
@output removed_count Number of removed elements.
@return true on success.
@error Propagates ENO from backing HashMap operations. }*)
function HashSet_remove_all: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: HashSetCoreType;
	end
	var_in_out
		elements: array[*] of HashSetElementType;
	end
	var_output
		removed_count: dint;
	end
	var_temp
		idx: dint;
		idx_begin, idx_end: dint;
		removed: bool;
	end
	{st}
	idx_begin := lower_bound(elements, 1);
	idx_end := upper_bound(elements, 1);

	for idx := idx_begin to idx_end by 1 do
		removed := HashSet_backing_remove(self:=self.backing, key:=elements[idx], eno=>eno);
		if not eno then
			return;
		end_if;
		if removed then
			removed_count := removed_count + 1;
		end_if;
	end_for;

	HashSet_remove_all := true;
	{end}
end

(*{doc Compare two HashSets for equality.
Require equal counts and mutual element containment.
@inout self First HashSet.
@inout other Second HashSet.
@return true when both contain the same elements. }*)
function HashSet_equals: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: HashSetCoreType;
		other: HashSetCoreType;
	end
	var_temp
		elements: array[HashSetCapacity] of HashSetElementType;
		written: dint;
		count_self: dint;
		count_other: dint;
		idx: dint;
		contained: bool;
	end
	{st}
	count_self := HashSet_backing_len(self:=self.backing, eno=>eno);
	if not eno then
		return;
	end_if;
	count_other := HashSet_backing_len(self:=other.backing, eno=>eno);
	if not eno then
		return;
	end_if;

	if count_self <> count_other then
		HashSet_equals := false;
		return;
	end_if;

	HashSet_backing_elements(self:=self.backing, keys:=elements, written=>written, eno=>eno);
	if not eno then
		return;
	end_if;

	for idx := 0 to written - 1 by 1 do
		contained := HashSet_backing_contains(self:=other.backing, key:=elements[idx], eno=>eno);
		if not eno then
			return;
		end_if;
		if not contained then
			HashSet_equals := false;
			return;
		end_if;
	end_for;

	HashSet_equals := true;
	{end}
end

{#ifdef HashSetElementHashFunction}
(*{doc Compute a hash value from all elements.
Accumulate element hashes in order-insensitive style.
@inout self Target HashSet.
@return Hash value. }*)
function HashSet_hash: udint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: HashSetCoreType;
	end
	var_temp
		elements: array[HashSetCapacity] of HashSetElementType;
		written: dint;
		idx: dint;
		elem_hash: udint;
		acc1: udint;
		acc2: udint;
	end
	{st}
	acc1 := udint#2166136261;
	acc2 := udint#0;

	HashSet_backing_elements(self:=self.backing, keys:=elements, written=>written, eno=>eno);
	if not eno then
		return;
	end_if;

	for idx := 0 to written - 1 by 1 do
		elem_hash := HashSetElementHashFunction(elements[idx]);
		acc1 := acc1 + elem_hash * udint#16777619 + udint#31;
		acc2 := acc2 + (elem_hash + udint#7) * (elem_hash + udint#13);
	end_for;

	HashSet_hash := acc1 * udint#2246822519 + acc2;
	{end}
end
{#endif}
