(*
	================================================================================
	OrderedSet Template Generator - OrderedMap-backed Insertion-Ordered Set
	================================================================================

	This template generates a typed OrderedSet from preprocessor macros.
	The internal implementation is based on OrderedMap (element as key, bool as value).

	Usage:
	1. Define required macros.
	2. Optionally define behavior macros.
	3. Include this template once per generated type.

	Required macros:
	  - OrderedSetName
	    Suffix used for generated types and functions.
	  - OrderedSetElementType
	    Element type stored in the set.
	  - OrderedSetElementHashFunction
	    Element hash function.
	  - OrderedSetCapacity
	    Number of table slots.

	Optional macros:
	  - OrderedSetMaxLoadPermille
	    Maximum load factor in permille.
	    Default: 800 (80%).
	  - OrderedSetElementEqualsFunction
	    Element equality function.
	    Default: direct value comparison.

	Generated APIs:
	  - init, copy, add, add_all
	  - remove, remove_all, contains, len
	  - elements, equals, hash, empty, clear

	Example (dint):
	  {#define OrderedSetName dint}
	  {#define OrderedSetCapacity 256}
	  {#define OrderedSetElementType dint}
	  {#define OrderedSetElementEqualsFunction Dint_equals}
	  {#define OrderedSetElementHashFunction Dint_hash}
	  {#include <container/orderedset/orderedset_template.txt>}

	Example (string):
	  {#define OrderedSetName str}
	  {#define OrderedSetCapacity 256}
	  {#define OrderedSetElementType string[128]}
	  {#define OrderedSetElementEqualsFunction String_equals}
	  {#define OrderedSetElementHashFunction String_hash}
	  {#include <container/orderedset/orderedset_template.txt>}

	Algorithm:
	  OrderedMap-backed set preserving insertion order.
	  equals and hash are insertion-order sensitive.
	================================================================================
*)

(*{doc Add one element.
Insert the element if not already present; preserve insertion order.
@inout self Target OrderedSet.
@input element Element to add.
@output added true when newly added, false when already present.
@return true on success, false on insert failure.
@error ENO is set to false on internal error. }*)
function OrderedSet_add: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedSetCoreType;
	end
	var_input
		element: OrderedSetElementType;
	end
	var_output
		added: bool;
	end
	var_temp
		replaced: bool;
	end
	{st}
	OrderedSet_add := OrderedSet_backing_upsert(self:=self.backing, key:=element, value:=true, replaced=>replaced, eno=>eno);
	if not eno then
		return;
	end_if;
	added := not replaced;
	{end}
end

(*{doc Add all elements from an input array.
Process input array in index order and count new insertions.
@inout self Target OrderedSet.
@inout elements Input elements.
@output added_count Number of newly added elements.
@return true on success, false if an insert fails.
@error ENO is propagated from backing OrderedMap. }*)
function OrderedSet_add_all: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedSetCoreType;
	end
	var_in_out
		elements: array[*] of OrderedSetElementType;
	end
	var_output
		added_count: dint;
	end
	var_temp
		idx: dint;
		idx_begin, idx_end: dint;
		replaced: bool;
		succeeded: bool;
	end
	{st}
	idx_begin := lower_bound(elements, 1);
	idx_end := upper_bound(elements, 1);
	added_count := 0;

	for idx := idx_begin to idx_end by 1 do
		succeeded := OrderedSet_backing_upsert(self:=self.backing, key:=elements[idx], value:=true, replaced=>replaced, eno=>eno);
		if not eno then
			return;
		end_if;
		if not succeeded then
			OrderedSet_add_all := false;
			return;
		end_if;
		if not replaced then
			added_count := added_count + 1;
		end_if;
	end_for;

	OrderedSet_add_all := true;
	{end}
end

(*{doc Remove all matching elements from an input array.
Process input array in index order and count successful deletions.
@inout self Target OrderedSet.
@inout elements Input elements.
@output removed_count Number of removed elements.
@return true on success.
@error Propagates ENO from backing OrderedMap operations. }*)
function OrderedSet_remove_all: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedSetCoreType;
	end
	var_in_out
		elements: array[*] of OrderedSetElementType;
	end
	var_output
		removed_count: dint;
	end
	var_temp
		idx: dint;
		idx_begin, idx_end: dint;
		removed: bool;
	end
	{st}
	idx_begin := lower_bound(elements, 1);
	idx_end := upper_bound(elements, 1);
	removed_count := 0;

	for idx := idx_begin to idx_end by 1 do
		removed := OrderedSet_backing_delete(self:=self.backing, key:=elements[idx], eno=>eno);
		if not eno then
			return;
		end_if;
		if removed then
			removed_count := removed_count + 1;
		end_if;
	end_for;

	OrderedSet_remove_all := true;
	{end}
end

(*{doc Compare two OrderedSets for equality.
Require same insertion order and same elements.
@inout self First OrderedSet.
@inout other Second OrderedSet.
@return true when both contain equal elements in equal insertion order.
@error ENO is set to false on internal error. }*)
function OrderedSet_equals: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedSetCoreType;
		other: OrderedSetCoreType;
	end
	var_temp
		idx: dint;
	end
	{st}
	if self.backing.order_count <> other.backing.order_count then
		OrderedSet_equals := false;
		return;
	end_if;

	for idx := 0 to self.backing.order_count - 1 by 1 do
		if not _ORDEREDSET_ELEM_EQUALS_(self.backing.order_keys[idx], other.backing.order_keys[idx]) then
			OrderedSet_equals := false;
			return;
		end_if;
	end_for;

	OrderedSet_equals := true;
	{end}
end

(*{doc Compute a hash value from all elements in insertion order.
Accumulate element hashes with sequence sensitivity.
@inout self Target OrderedSet.
@return Hash value. }*)
function OrderedSet_hash: udint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedSetCoreType;
	end
	var_temp
		acc: udint;
		idx: dint;
		elem_hash: udint;
	end
	{st}
	acc := udint#2166136261;
	for idx := 0 to self.backing.order_count - 1 by 1 do
		elem_hash := OrderedSetElementHashFunction(self.backing.order_keys[idx]);
		acc := dword_to_udint(udint_to_dword(acc) xor udint_to_dword(elem_hash * udint#16777619)) * udint#16777619 + dint_to_udint(idx + 1);
	end_for;
	OrderedSet_hash := acc + dint_to_udint(self.backing.order_count);
	{end}
end
