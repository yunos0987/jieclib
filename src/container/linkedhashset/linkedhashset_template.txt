(*
	================================================================================
	LinkedHashSet Template Generator - LinkedHashMap-backed Linked-Order Set
	================================================================================

	This template generates a typed LinkedHashSet from preprocessor macros.
	The internal implementation is based on LinkedHashMap (element as key, bool as value).

	Usage:
	1. Define required macros.
	2. Optionally define behavior macros.
	3. Include this template once per generated type.

	Required macros:
	  - LinkedHashSetName
	    Suffix used for generated types and functions.
	  - LinkedHashSetElementType
	    Element type stored in the set.
	  - LinkedHashSetElementHashFunction
	    Element hash function.
	  - LinkedHashSetCapacity
	    Number of table slots.

	Optional macros:
	  - LinkedHashSetMaxLoadPermille
	    Maximum load factor in permille.
	    Default: 800 (80%).
	  - LinkedHashSetElementEqualsFunction
	    Element equality function.
	    Default: direct value comparison.
	  - LinkedHashSetAccessOrder
	    When true, add of an existing element moves it to tail (LRU behavior).
	    Default: false.

	Generated APIs:
	  - init, copy, add, add_all
	  - remove, remove_all, contains, len
	  - elements, equals, hash, empty, clear

	Example (dint):
	  {#define LinkedHashSetName dint}
	  {#define LinkedHashSetCapacity 256}
	  {#define LinkedHashSetElementType dint}
	  {#define LinkedHashSetElementEqualsFunction Dint_equals}
	  {#define LinkedHashSetElementHashFunction Dint_hash}
	  {#include <container/linkedhashset/linkedhashset_template.txt>}

	Example (string):
	  {#define LinkedHashSetName str}
	  {#define LinkedHashSetCapacity 256}
	  {#define LinkedHashSetElementType string[128]}
	  {#define LinkedHashSetElementEqualsFunction String_equals}
	  {#define LinkedHashSetElementHashFunction String_hash}
	  {#include <container/linkedhashset/linkedhashset_template.txt>}

	Algorithm:
	  LinkedHashMap-backed set preserving insertion order via doubly linked nodes.
	  equals and hash are linked-order sensitive.
	================================================================================
*)

	{end}
end

	{end}
end

(*{doc Add one element.
Insert the element if not already present; preserve linked order.
In access-order mode, re-adding an existing element moves it to the tail.
@inout self Target LinkedHashSet.
@input element Element to add.
@output added true when newly added, false when already present.
@return true on success, false on insert failure.
@error ENO is set to false on internal error. }*)
function LinkedHashSet_add: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashSetCoreType;
	end
	var_input
		element: LinkedHashSetElementType;
	end
	var_output
		added: bool;
	end
	var_temp
		replaced: bool;
	end
	{st}
	LinkedHashSet_add := LinkedHashSet_backing_upsert(self:=self.backing, key:=element, value:=true, replaced=>replaced, eno=>eno);
	if not eno then
		return;
	end_if;
	added := not replaced;
	{end}
end

(*{doc Add all elements from an input array.
Process input array in index order and count new insertions.
@inout self Target LinkedHashSet.
@inout elements Input elements.
@output added_count Number of newly added elements.
@return true on success, false if an insert fails.
@error ENO is propagated from backing LinkedHashMap. }*)
function LinkedHashSet_add_all: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashSetCoreType;
	end
	var_in_out
		elements: array[*] of LinkedHashSetElementType;
	end
	var_output
		added_count: dint;
	end
	var_temp
		idx: dint;
		idx_begin, idx_end: dint;
		replaced: bool;
		succeeded: bool;
	end
	{st}
	idx_begin := lower_bound(elements, 1);
	idx_end := upper_bound(elements, 1);
	added_count := 0;

	for idx := idx_begin to idx_end by 1 do
		succeeded := LinkedHashSet_backing_upsert(self:=self.backing, key:=elements[idx], value:=true, replaced=>replaced, eno=>eno);
		if not eno then
			return;
		end_if;
		if not succeeded then
			LinkedHashSet_add_all := false;
			return;
		end_if;
		if not replaced then
			added_count := added_count + 1;
		end_if;
	end_for;

	LinkedHashSet_add_all := true;
	{end}
end

(*{doc Compare two LinkedHashSets for equality.
Require same count, same elements in same linked order.
@inout self First LinkedHashSet.
@inout other Second LinkedHashSet.
@return true when both contain equal elements in equal linked order.
@error ENO is set to false on internal error. }*)
function LinkedHashSet_equals: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashSetCoreType;
		other: LinkedHashSetCoreType;
	end
	var_temp
		cur_self: dint;
		cur_other: dint;
	end
	{st}
	if self.backing.count <> other.backing.count then
		LinkedHashSet_equals := false;
		return;
	end_if;

	cur_self := self.backing.head_idx;
	cur_other := other.backing.head_idx;
	while (cur_self >= 0) and (cur_other >= 0) do
		if (cur_self >= LinkedHashSetCapacity) or (not self.backing.nodes[cur_self].used) then
			eno := false;
			return;
		end_if;
		if (cur_other >= LinkedHashSetCapacity) or (not other.backing.nodes[cur_other].used) then
			eno := false;
			return;
		end_if;

		if not _LINKEDHASHSET_ELEM_EQUALS_(self.backing.nodes[cur_self].key, other.backing.nodes[cur_other].key) then
			LinkedHashSet_equals := false;
			return;
		end_if;

		cur_self := self.backing.nodes[cur_self].next;
		cur_other := other.backing.nodes[cur_other].next;
	end_while;

	if (cur_self <> -1) or (cur_other <> -1) then
		eno := false;
		return;
	end_if;

	LinkedHashSet_equals := true;
	{end}
end

(*{doc Compute a hash value from all elements in linked order.
Accumulate element hashes with sequence sensitivity.
@inout self Target LinkedHashSet.
@return Hash value. }*)
function LinkedHashSet_hash: udint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashSetCoreType;
	end
	var_temp
		acc: udint;
		elem_hash: udint;
		cur: dint;
		sequence_idx: dint;
	end
	{st}
	acc := udint#2166136261;
	cur := self.backing.head_idx;
	sequence_idx := 0;

	while cur >= 0 do
		if (cur >= LinkedHashSetCapacity) or (not self.backing.nodes[cur].used) then
			eno := false;
			return;
		end_if;
		elem_hash := LinkedHashSetElementHashFunction(self.backing.nodes[cur].key);
		acc := dword_to_udint(udint_to_dword(acc) xor udint_to_dword(elem_hash)) * udint#16777619 + elem_hash + dint_to_udint(sequence_idx + 1);
		cur := self.backing.nodes[cur].next;
		sequence_idx := sequence_idx + 1;
	end_while;

	LinkedHashSet_hash := acc + dint_to_udint(self.backing.count);
	{end}
end
