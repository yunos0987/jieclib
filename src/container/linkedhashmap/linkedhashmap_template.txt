(*
	================================================================================
	LinkedHashMap Template Generator - HashMap + Doubly Linked Order
	================================================================================

	This template generates a typed LinkedHashMap from preprocessor macros.
	The internal implementation is a HashMap-backed doubly linked structure.

	Usage:
	1. Define required macros.
	2. Optionally define behavior macros.
	3. Include this template once per generated type.

	Required macros:
	  - LinkedHashMapName
	    Suffix used for generated types and functions.
	  - LinkedHashMapKeyType
	    Key type.
	  - LinkedHashMapValueType
	    Value type.
	  - LinkedHashMapKeyEqualsFunction
	    Key equality function.
	  - LinkedHashMapKeyHashFunction
	    Key hash function.
	  - LinkedHashMapCapacity
	    Number of table slots.

	Optional macros:
	  - LinkedHashMapMaxLoadPermille
	    Maximum load factor in permille.
	    Default: 800 (80%).
	  - LinkedHashMapAccessOrder
	    When true, successful lookup/upsert moves key to tail.
	    Default: false.
	  - LinkedHashMapValueEqualsFunction
	    Value equality function for equals.
	    Default: direct value comparison.

	Generated APIs:
	  - init, copy, upsert, lookup, delete
	  - contains, len, keys, values, items
	  - equals, hash, empty, clear

	Example (string):
	  {#define LinkedHashMapName str}
	  {#define LinkedHashMapCapacity 256}
	  {#define LinkedHashMapKeyType string[128]}
	  {#define LinkedHashMapValueType dint}
	  {#define LinkedHashMapKeyEqualsFunction String_equals}
	  {#define LinkedHashMapKeyHashFunction String_hash}
	  {#include <container/linkedhashmap/linkedhashmap_template.txt>}

	Example (dint):
	  {#define LinkedHashMapName dint}
	  {#define LinkedHashMapCapacity 256}
	  {#define LinkedHashMapKeyType dint}
	  {#define LinkedHashMapValueType dint}
	  {#define LinkedHashMapKeyEqualsFunction Dint_equals}
	  {#define LinkedHashMapKeyHashFunction Dint_hash}
	  {#define LinkedHashMapAccessOrder true}
	  {#include <container/linkedhashmap/linkedhashmap_template.txt>}

	Algorithm:
	  HashMap-backed index map plus doubly linked order list.
	================================================================================
*)

type
	LinkedHashMapItemType: struct
		key: LinkedHashMapKeyType;
		value: LinkedHashMapValueType;
	end
end

type
	LinkedHashMapNodeType: struct
		used: bool;
		key: LinkedHashMapKeyType;
		value: LinkedHashMapValueType;
		prev: dint;
		next: dint;
	end
end

type
	LinkedHashMapCoreType: struct
		index_map: LinkedHashMap_backing_type;
		nodes: array[LinkedHashMapCapacity] of LinkedHashMapNodeType;
		free_stack: array[LinkedHashMapCapacity] of dint;
		free_top: dint;
		head_idx: dint;
		tail_idx: dint;
		count: dint;
	end
end

(*{doc Initialize the container.
Reset linked-order nodes, free stack, and backing map.
@inout self Target LinkedHashMap. }*)
function LinkedHashMap_init
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashMapCoreType;
	end
	var_temp
		idx: dint;
	end
	{st}
	LinkedHashMap_backing_init(self:=self.index_map);

	for idx := 0 to LinkedHashMapCapacity - 1 by 1 do
		self.nodes[idx].used := false;
		self.nodes[idx].prev := -1;
		self.nodes[idx].next := -1;
		self.free_stack[idx] := idx;
	end_for;
	self.free_top := LinkedHashMapCapacity;
	self.head_idx := -1;
	self.tail_idx := -1;
	self.count := 0;
	{end}
end

(*{doc Copy container state.
Rebuild destination by ordered upsert from source.
@inout self Source LinkedHashMap.
@inout dest Destination LinkedHashMap.
@return true on success.
@error ENO is set to false on internal error. }*)
function LinkedHashMap_copy: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashMapCoreType;
		dest: LinkedHashMapCoreType;
	end
	var_temp
		cur: dint;
		replaced: bool;
	end
	{st}
	LinkedHashMap_init(dest);

	cur := self.head_idx;
	while cur >= 0 do
		if (cur >= LinkedHashMapCapacity) or (not self.nodes[cur].used) then
			eno := false;
			return;
		end_if;
		if not LinkedHashMap_upsert(self:=dest, key:=self.nodes[cur].key, value:=self.nodes[cur].value, replaced=>replaced, eno=>eno) then
			if not eno then
				return;
			end_if;
			LinkedHashMap_copy := false;
			return;
		end_if;
		cur := self.nodes[cur].next;
	end_while;

	LinkedHashMap_copy := true;
	{end}
end

(*{doc Insert or update a key/value pair.
Insert new node in linked order or update existing node value.
@inout self Target LinkedHashMap.
@input key Key.
@input value Value.
@output replaced true when updated, false when inserted.
@return true on success, false on failure.
@error ENO is set to false on internal error. }*)
function LinkedHashMap_upsert: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashMapCoreType;
	end
	var_input
		key: LinkedHashMapKeyType;
		value: LinkedHashMapValueType;
	end
	var_output
		replaced: bool;
	end
	var_temp
		idx: dint;
		prev_idx: dint;
		next_idx: dint;
		replaced_internal: bool;
		succeeded: bool;
	end
	{st}
	LinkedHashMap_backing_lookup(self:=self.index_map, key:=key, value:=idx, eno=>eno);
	if eno then
		if (idx < 0) or (idx >= LinkedHashMapCapacity) or (not self.nodes[idx].used) then
			eno := false;
			return;
		end_if;

		self.nodes[idx].value := value;

		if LinkedHashMapAccessOrder and (idx <> self.tail_idx) then
			prev_idx := self.nodes[idx].prev;
			next_idx := self.nodes[idx].next;

			if prev_idx >= 0 then
				self.nodes[prev_idx].next := next_idx;
			else
				self.head_idx := next_idx;
			end_if;
			if next_idx >= 0 then
				self.nodes[next_idx].prev := prev_idx;
			else
				self.tail_idx := prev_idx;
			end_if;

			self.nodes[idx].prev := self.tail_idx;
			self.nodes[idx].next := -1;
			if self.tail_idx >= 0 then
				self.nodes[self.tail_idx].next := idx;
			else
				self.head_idx := idx;
			end_if;
			self.tail_idx := idx;
		end_if;

		replaced := true;
		LinkedHashMap_upsert := true;
		return;
	end_if;

	// lookup eno=false ,
	eno := true;

	if self.free_top <= 0 then
		replaced := false;
		LinkedHashMap_upsert := false;
		return;
	end_if;

	self.free_top := self.free_top - 1;
	idx := self.free_stack[self.free_top];
	if (idx < 0) or (idx >= LinkedHashMapCapacity) then
		eno := false;
		return;
	end_if;

	self.nodes[idx].used := true;
	self.nodes[idx].key := key;
	self.nodes[idx].value := value;
	self.nodes[idx].prev := self.tail_idx;
	self.nodes[idx].next := -1;
	if self.tail_idx >= 0 then
		self.nodes[self.tail_idx].next := idx;
	else
		self.head_idx := idx;
	end_if;
	self.tail_idx := idx;

	succeeded := LinkedHashMap_backing_upsert(self:=self.index_map, key:=key, value:=idx, replaced=>replaced_internal, eno=>eno);
	if (not eno) or (not succeeded) or replaced_internal then
		// rollback
		prev_idx := self.nodes[idx].prev;
		if prev_idx >= 0 then
			self.nodes[prev_idx].next := -1;
		else
			self.head_idx := -1;
		end_if;
		self.tail_idx := prev_idx;
		self.nodes[idx].used := false;
		self.nodes[idx].prev := -1;
		self.nodes[idx].next := -1;
		self.free_stack[self.free_top] := idx;
		self.free_top := self.free_top + 1;
		if replaced_internal then
			eno := false;
		end_if;
		if not eno then
			return;
		end_if;
		replaced := false;
		LinkedHashMap_upsert := false;
		return;
	end_if;

	self.count := self.count + 1;
	replaced := false;
	LinkedHashMap_upsert := true;
	{end}
end

(*{doc Lookup a value by key.
Read value and optionally update access order.
@inout self Target LinkedHashMap.
@input key Target key.
@inout value Output value.
@error ENO is set to false when key is not found or on internal error. }*)
function LinkedHashMap_lookup
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashMapCoreType;
	end
	var_input
		key: LinkedHashMapKeyType;
	end
	var_in_out
		value: LinkedHashMapValueType;
	end
	var_temp
		idx: dint;
		prev_idx: dint;
		next_idx: dint;
	end
	{st}
	LinkedHashMap_backing_lookup(self:=self.index_map, key:=key, value:=idx, eno=>eno);
	if not eno then
		return;
	end_if;
	if (idx < 0) or (idx >= LinkedHashMapCapacity) or (not self.nodes[idx].used) then
		eno := false;
		return;
	end_if;

	value := self.nodes[idx].value;

	if LinkedHashMapAccessOrder and (idx <> self.tail_idx) then
		prev_idx := self.nodes[idx].prev;
		next_idx := self.nodes[idx].next;

		if prev_idx >= 0 then
			self.nodes[prev_idx].next := next_idx;
		else
			self.head_idx := next_idx;
		end_if;
		if next_idx >= 0 then
			self.nodes[next_idx].prev := prev_idx;
		else
			self.tail_idx := prev_idx;
		end_if;

		self.nodes[idx].prev := self.tail_idx;
		self.nodes[idx].next := -1;
		if self.tail_idx >= 0 then
			self.nodes[self.tail_idx].next := idx;
		else
			self.head_idx := idx;
		end_if;
		self.tail_idx := idx;
	end_if;
	{end}
end

(*{doc Delete a key/value pair.
Remove from both backing map and linked-order chain.
@inout self Target LinkedHashMap.
@input key Target key to delete.
@return true when deleted, false when key is not found.
@error ENO is set to false on internal error. }*)
function LinkedHashMap_delete: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashMapCoreType;
	end
	var_input
		key: LinkedHashMapKeyType;
	end
	var_temp
		idx: dint;
		prev_idx: dint;
		next_idx: dint;
		removed: bool;
	end
	{st}
	LinkedHashMap_backing_lookup(self:=self.index_map, key:=key, value:=idx, eno=>eno);
	if not eno then
		eno := true;
		LinkedHashMap_delete := false;
		return;
	end_if;
	if (idx < 0) or (idx >= LinkedHashMapCapacity) or (not self.nodes[idx].used) then
		eno := false;
		return;
	end_if;

	removed := LinkedHashMap_backing_delete(self:=self.index_map, key:=key, eno=>eno);
	if not eno then
		return;
	end_if;
	if not removed then
		eno := false;
		return;
	end_if;

	prev_idx := self.nodes[idx].prev;
	next_idx := self.nodes[idx].next;
	if prev_idx >= 0 then
		self.nodes[prev_idx].next := next_idx;
	else
		self.head_idx := next_idx;
	end_if;
	if next_idx >= 0 then
		self.nodes[next_idx].prev := prev_idx;
	else
		self.tail_idx := prev_idx;
	end_if;

	self.nodes[idx].used := false;
	self.nodes[idx].prev := -1;
	self.nodes[idx].next := -1;
	if self.free_top >= LinkedHashMapCapacity then
		eno := false;
		return;
	end_if;
	self.free_stack[self.free_top] := idx;
	self.free_top := self.free_top + 1;
	self.count := self.count - 1;

	LinkedHashMap_delete := true;
	{end}
end

(*{doc Copy keys into an output array in linked order.
Export keys from head to tail.
@inout self Target LinkedHashMap.
@inout keys Output array.
@output written Number of copied keys.
@return true on success.
@error ENO is set to false on invalid state or insufficient output capacity. }*)
function LinkedHashMap_keys: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashMapCoreType;
	end
	var_in_out
		keys: array[*] of LinkedHashMapKeyType;
	end
	var_output
		written: dint;
	end
	var_temp
		out_idx: dint;
		out_hi: dint;
		cur: dint;
	end
	{st}
	out_idx := lower_bound(keys, 1);
	out_hi := upper_bound(keys, 1);
	written := 0;
	cur := self.head_idx;

	while cur >= 0 do
		if (cur >= LinkedHashMapCapacity) or (not self.nodes[cur].used) then
			eno := false;
			return;
		end_if;
		if out_idx > out_hi then
			eno := false;
			return;
		end_if;
		keys[out_idx] := self.nodes[cur].key;
		out_idx := out_idx + 1;
		written := written + 1;
		cur := self.nodes[cur].next;
	end_while;

	LinkedHashMap_keys := true;
	{end}
end

		values[out_idx] := self.nodes[cur].value;
		out_idx := out_idx + 1;
		written := written + 1;
		cur := self.nodes[cur].next;
	end_while;

	LinkedHashMap_values := true;
	{end}
end

(*{doc Copy key/value items into an output array in linked order.
Export items from head to tail.
@inout self Target LinkedHashMap.
@inout items Output array.
@output written Number of copied items.
@return true on success.
@error ENO is set to false on invalid state or insufficient capacity. }*)
function LinkedHashMap_items: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashMapCoreType;
	end
	var_in_out
		items: array[*] of LinkedHashMapItemType;
	end
	var_output
		written: dint;
	end
	var_temp
		out_idx: dint;
		out_hi: dint;
		cur: dint;
	end
	{st}
	out_idx := lower_bound(items, 1);
	out_hi := upper_bound(items, 1);
	written := 0;
	cur := self.head_idx;

	while cur >= 0 do
		if (cur >= LinkedHashMapCapacity) or (not self.nodes[cur].used) then
			eno := false;
			return;
		end_if;
		if out_idx > out_hi then
			eno := false;
			return;
		end_if;
		items[out_idx].key := self.nodes[cur].key;
		items[out_idx].value := self.nodes[cur].value;
		out_idx := out_idx + 1;
		written := written + 1;
		cur := self.nodes[cur].next;
	end_while;

	LinkedHashMap_items := true;
	{end}
end

(*{doc Compare two LinkedHashMaps for equality.
Require same size, key order, and values.
@inout self First LinkedHashMap.
@inout other Second LinkedHashMap.
@return true when order, keys, and values all match.
@error ENO is set to false on internal error. }*)
function LinkedHashMap_equals: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashMapCoreType;
		other: LinkedHashMapCoreType;
	end
	var_temp
		cur_a: dint;
		cur_b: dint;
	end
	{st}
	if self.count <> other.count then
		LinkedHashMap_equals := false;
		return;
	end_if;

	cur_a := self.head_idx;
	cur_b := other.head_idx;
	while (cur_a >= 0) and (cur_b >= 0) do
		if (cur_a >= LinkedHashMapCapacity) or (not self.nodes[cur_a].used) then
			eno := false;
			return;
		end_if;
		if (cur_b >= LinkedHashMapCapacity) or (not other.nodes[cur_b].used) then
			eno := false;
			return;
		end_if;

		if not LinkedHashMapKeyEqualsFunction(self.nodes[cur_a].key, other.nodes[cur_b].key) then
			LinkedHashMap_equals := false;
			return;
		end_if;
		if not _LINKEDHASHMAP_VALUE_EQUALS_(self.nodes[cur_a].value, other.nodes[cur_b].value) then
			LinkedHashMap_equals := false;
			return;
		end_if;

		cur_a := self.nodes[cur_a].next;
		cur_b := other.nodes[cur_b].next;
	end_while;

	if (cur_a <> -1) or (cur_b <> -1) then
		eno := false;
		return;
	end_if;

	LinkedHashMap_equals := true;
	{end}
end

{#ifdef LinkedHashMapValueHashFunction}
(*{doc Compute a hash value from all key/value pairs in linked order.
Accumulate key/value hashes with sequence sensitivity.
@inout self Target LinkedHashMap.
@return Hash value. }*)
function LinkedHashMap_hash: udint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashMapCoreType;
	end
	var_temp
		acc: udint;
		key_hash, value_hash: udint;
		cur: dint;
		sequence_idx: dint;
	end
	{st}
	acc := udint#2166136261;
	cur := self.head_idx;
	sequence_idx := 0;

	while cur >= 0 do
		if (cur >= LinkedHashMapCapacity) or (not self.nodes[cur].used) then
			eno := false;
			return;
		end_if;
		key_hash := LinkedHashMapKeyHashFunction(self.nodes[cur].key);
		value_hash := LinkedHashMapValueHashFunction(self.nodes[cur].value);
		acc := dword_to_udint(udint_to_dword(acc) xor udint_to_dword(key_hash)) * udint#16777619 + value_hash + dint_to_udint(sequence_idx + 1);
		cur := self.nodes[cur].next;
		sequence_idx := sequence_idx + 1;
	end_while;

	LinkedHashMap_hash := acc + dint_to_udint(self.count);
	{end}
end
{#endif}

(*{doc Clear all entries.
Reset backing map, linked nodes, and free stack.
@inout self Target LinkedHashMap.
@return true on success.
@error ENO is set to false on internal error. }*)
function LinkedHashMap_clear: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: LinkedHashMapCoreType;
	end
	var_temp
		idx: dint;
	end
	{st}
	if not LinkedHashMap_backing_clear(self:=self.index_map, eno=>eno) then
		if not eno then
			return;
		end_if;
		LinkedHashMap_clear := false;
		return;
	end_if;

	for idx := 0 to LinkedHashMapCapacity - 1 by 1 do
		self.nodes[idx].used := false;
		self.nodes[idx].prev := -1;
		self.nodes[idx].next := -1;
		self.free_stack[idx] := idx;
	end_for;
	self.free_top := LinkedHashMapCapacity;
	self.head_idx := -1;
	self.tail_idx := -1;
	self.count := 0;

	LinkedHashMap_clear := true;
	{end}
end
