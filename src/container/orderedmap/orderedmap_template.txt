(*
	================================================================================
	OrderedMap Template Generator - HashMap + Insertion Order
	================================================================================

	This template generates a typed OrderedMap from preprocessor macros.
	The internal implementation is based on HashMap plus insertion-order keys.

	Usage:
	1. Define required macros.
	2. Optionally define behavior macros.
	3. Include this template once per generated type.

	Required macros:
	  - OrderedMapName
	    Suffix used for generated types and functions.
	  - OrderedMapKeyType
	    Key type.
	  - OrderedMapValueType
	    Value type.
	  - OrderedMapKeyEqualsFunction
	    Key equality function.
	  - OrderedMapKeyHashFunction
	    Key hash function.
	  - OrderedMapCapacity
	    Number of table slots.

	Optional macros:
	  - OrderedMapMaxLoadPermille
	    Maximum load factor in permille.
	    Default: 800 (80%).
	  - OrderedMapValueEqualsFunction
	    Value equality function for equals.
	    Default: direct value comparison.

	Generated APIs:
	  - init, copy, upsert, lookup, delete
	  - contains, len, keys, values, items
	  - equals, hash, empty, clear

	Example (string):
	  {#define OrderedMapName str}
	  {#define OrderedMapCapacity 256}
	  {#define OrderedMapKeyType string[128]}
	  {#define OrderedMapValueType dint}
	  {#define OrderedMapKeyEqualsFunction String_equals}
	  {#define OrderedMapKeyHashFunction String_hash}
	  {#include <container/orderedmap/orderedmap_template.txt>}

	Example (dint):
	  {#define OrderedMapName dint}
	  {#define OrderedMapCapacity 256}
	  {#define OrderedMapKeyType dint}
	  {#define OrderedMapValueType dint}
	  {#define OrderedMapKeyEqualsFunction Dint_equals}
	  {#define OrderedMapKeyHashFunction Dint_hash}
	  {#include <container/orderedmap/orderedmap_template.txt>}

	Algorithm:
	  HashMap-backed ordered key list preserving insertion order.
	================================================================================
*)

type
	(* OrderedMap key/value item type. *)
	OrderedMapItemType: struct
		(* Item key. *)
		key: OrderedMapKeyType;
		(* Item value. *)
		value: OrderedMapValueType;
	end
end

type
	(* OrderedMap runtime state. *)
	OrderedMapCoreType: struct
		(* Backing HashMap for key/value storage. *)
		map: OrderedMap_backing_type;
		(* Keys in insertion order. *)
		order_keys: array[OrderedMapCapacity] of OrderedMapKeyType;
		(* Number of keys in insertion order list. *)
		order_count: dint;
	end
end

(*{doc Initialize the container.
Reset insertion order state and backing map.
@inout self Target OrderedMap. }*)
function OrderedMap_init
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedMapCoreType;
	end
	{st}
	OrderedMap_backing_init(self:=self.map);

	self.order_count := 0;
	{end}
end

(*{doc Copy container state.
Copy backing map and insertion order list.
@inout self Source OrderedMap.
@inout dest Destination OrderedMap.
@return true on success.
@error Propagates ENO from backing HashMap operations. }*)
function OrderedMap_copy: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedMapCoreType;
		dest: OrderedMapCoreType;
	end
	var_temp
		idx: dint;
	end
	{st}
	if not OrderedMap_backing_copy(self:=self.map, dest:=dest.map) then
		OrderedMap_copy := false;
		return;
	end_if;

	dest.order_count := self.order_count;
	for idx := 0 to OrderedMapCapacity - 1 by 1 do
		dest.order_keys[idx] := self.order_keys[idx];
	end_for;

	OrderedMap_copy := true;
	{end}
end

(*{doc Insert or update a key/value pair.
Append key to order list only on first insertion.
@inout self Target OrderedMap.
@input key Key.
@input value Value.
@output replaced true when updated, false when inserted.
@return true on success, false on insert failure.
@error ENO is set to false on internal error. }*)
function OrderedMap_upsert: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedMapCoreType;
	end
	var_input
		key: OrderedMapKeyType;
		value: OrderedMapValueType;
	end
	var_output
		replaced: bool;
	end
	{st}
	if not OrderedMap_backing_upsert(self:=self.map, key:=key, value:=value, replaced=>replaced, eno=>eno) then
		if not eno then
			return;
		end_if;
		OrderedMap_upsert := false;
		return;
	end_if;

	if not replaced then
		if self.order_count >= OrderedMapCapacity then
			eno := false;
			return;
		end_if;
		self.order_keys[self.order_count] := key;
		self.order_count := self.order_count + 1;
	end_if;

	OrderedMap_upsert := true;
	{end}
end

(*{doc Delete a key/value pair.
Remove key from both backing map and insertion-order list.
@inout self Target OrderedMap.
@input key Target key to delete.
@return true when deleted, false when key is not found.
@error ENO is set to false on internal error. }*)
function OrderedMap_delete: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedMapCoreType;
	end
	var_input
		key: OrderedMapKeyType;
	end
	var_temp
		found_idx: dint;
		idx: dint;
	end
	{st}
	if not OrderedMap_backing_delete(self:=self.map, key:=key, eno=>eno) then
		if not eno then
			return;
		end_if;
		OrderedMap_delete := false;
		return;
	end_if;

	found_idx := -1;
	for idx := 0 to self.order_count - 1 by 1 do
		if OrderedMapKeyEqualsFunction(self.order_keys[idx], key) then
			found_idx := idx;
			exit;
		end_if;
	end_for;
	if found_idx < 0 then
		eno := false;
		return;
	end_if;

	for idx := found_idx to self.order_count - 2 by 1 do
		self.order_keys[idx] := self.order_keys[idx + 1];
	end_for;
	self.order_count := self.order_count - 1;

	OrderedMap_delete := true;
	{end}
end

(*{doc Copy keys into an output array in insertion order.
Export keys from oldest to newest insertion.
@inout self Target OrderedMap.
@inout keys Output array.
@output written Number of copied keys.
@return true on success.
@error ENO is set to false when output capacity is insufficient. }*)
function OrderedMap_keys: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedMapCoreType;
	end
	var_in_out
		keys: array[*] of OrderedMapKeyType;
	end
	var_output
		written: dint;
	end
	var_temp
		out_idx: dint;
		out_hi: dint;
		idx: dint;
	end
	{st}
	out_idx := lower_bound(keys, 1);
	out_hi := upper_bound(keys, 1);
	written := 0;

	for idx := 0 to self.order_count - 1 by 1 do
		if out_idx > out_hi then
			eno := false;
			return;
		end_if;
		keys[out_idx] := self.order_keys[idx];
		out_idx := out_idx + 1;
		written := written + 1;
	end_for;

	OrderedMap_keys := true;
	{end}
end

(*{doc Copy values into an output array in insertion order.
Resolve each value by ordered key traversal.
@inout self Target OrderedMap.
@inout values Output array.
@output written Number of copied values.
@return true on success.
@error ENO is set to false on internal error or insufficient output capacity. }*)
function OrderedMap_values: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedMapCoreType;
	end
	var_in_out
		values: array[*] of OrderedMapValueType;
	end
	var_output
		written: dint;
	end
	var_temp
		out_idx: dint;
		out_hi: dint;
		idx: dint;
		value: OrderedMapValueType;
	end
	{st}
	out_idx := lower_bound(values, 1);
	out_hi := upper_bound(values, 1);
	written := 0;

	for idx := 0 to self.order_count - 1 by 1 do
		if out_idx > out_hi then
			eno := false;
			return;
		end_if;
		OrderedMap_backing_lookup(self:=self.map, key:=self.order_keys[idx], value:=value, eno=>eno);
		if not eno then
			return;
		end_if;
		values[out_idx] := value;
		out_idx := out_idx + 1;
		written := written + 1;
	end_for;

	OrderedMap_values := true;
	{end}
end

(*{doc Compare two OrderedMaps for equality.
Require same insertion order and same values per key.
@inout self First OrderedMap.
@inout other Second OrderedMap.
@return true when insertion order and all key/value pairs match.
@error ENO is set to false on internal error. }*)
function OrderedMap_equals: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedMapCoreType;
		other: OrderedMapCoreType;
	end
	var_temp
		idx: dint;
		v_self: OrderedMapValueType;
		v_other: OrderedMapValueType;
	end
	{st}
	if self.order_count <> other.order_count then
		OrderedMap_equals := false;
		return;
	end_if;

	for idx := 0 to self.order_count - 1 by 1 do
		if not OrderedMapKeyEqualsFunction(self.order_keys[idx], other.order_keys[idx]) then
			OrderedMap_equals := false;
			return;
		end_if;

		OrderedMap_backing_lookup(self:=self.map, key:=self.order_keys[idx], value:=v_self, eno=>eno);
		if not eno then
			return;
		end_if;

		OrderedMap_backing_lookup(self:=other.map, key:=other.order_keys[idx], value:=v_other, eno=>eno);
		if not eno then
			return;
		end_if;
		if not _ORDEREDMAP_VALUE_EQUALS_(v_self, v_other) then
			OrderedMap_equals := false;
			return;
		end_if;
	end_for;

	OrderedMap_equals := true;
	{end}
end

{#ifdef OrderedMapValueHashFunction}
(*{doc Compute a hash value from all key/value pairs in insertion order.
Accumulate key/value hashes with sequence sensitivity.
@inout self Target OrderedMap.
@return Hash value. }*)
function OrderedMap_hash: udint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: OrderedMapCoreType;
	end
	var_temp
		acc: udint;
		idx: dint;
		key_hash, value_hash: udint;
		value: OrderedMapValueType;
	end
	{st}
	acc := udint#2166136261;
	for idx := 0 to self.order_count - 1 by 1 do
		key_hash := OrderedMapKeyHashFunction(self.order_keys[idx]);
		OrderedMap_backing_lookup(self:=self.map, key:=self.order_keys[idx], value:=value, eno=>eno);
		if not eno then
			return;
		end_if;
		value_hash := OrderedMapValueHashFunction(value);
		acc := dword_to_udint(udint_to_dword(acc) xor udint_to_dword(key_hash + value_hash * udint#16777619)) * udint#16777619 + dint_to_udint(idx + 1);
	end_for;
	OrderedMap_hash := acc + dint_to_udint(self.order_count);
	{end}
end
{#endif}
