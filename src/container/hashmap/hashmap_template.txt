(*
	================================================================================
	HashMap Template Generator - Macro-based Generic HashMap
	================================================================================

	This template generates a typed HashMap from preprocessor macros.

	Usage:
	1. Define required macros.
	2. Optionally define behavior macros.
	3. Include this template once per generated type.

	Required macros:
	  - HashMapName
	    Suffix used for generated types and functions.
	  - HashMapKeyType
	    Key type.
	  - HashMapValueType
	    Value type.
	  - HashMapKeyEqualsFunction
	    Key equality function.
	  - HashMapKeyHashFunction
	    Key hash function.
	  - HashMapCapacity
	    Number of table slots.

	Optional macros:
	  - HashMapMaxLoadPermille
	    Maximum load factor in permille.
	    Default: 800 (80%).
	  - HashMapValueEqualsFunction
	    Value equality function for equals.
	    Default: direct value comparison.

	Generated APIs:
	  - init, copy, upsert, lookup, delete
	  - contains, len, keys, values, items
	  - equals, hash, empty, clear

	Example (string):
	  {#define HashMapName str}
	  {#define HashMapCapacity 256}
	  {#define HashMapValueType dint}
	  {#define HashMapKeyType string[128]}
	  {#define HashMapKeyEqualsFunction String_equals}
	  {#define HashMapKeyHashFunction String_hash}
	  {#include <container/hashmap/hashmap_template.txt>}

	Example (dint):
	  {#define HashMapName dint}
	  {#define HashMapCapacity 256}
	  {#define HashMapValueType dint}
	  {#define HashMapKeyType dint}
	  {#define HashMapKeyEqualsFunction Dint_equals}
	  {#define HashMapKeyHashFunction Dint_hash}
	  {#include <container/hashmap/hashmap_template.txt>}

	Algorithm:
	  Robin Hood hashing with probe-distance insertion and backward-shift delete.
	================================================================================
*)

type

(*{doc Compute probe distance for Robin Hood hashing.
Calculate wrapped distance from ideal slot to current slot.
@input idx Current slot index.
@input ideal_idx Ideal slot index.
@input capacity Table capacity.
@return Probe distance in range [0, capacity).
@error ENO is set to false on invalid input. }*)
function HashMap_private_static_robinhood_probedist: dint
	_PROLOGUE_OF_FUNCTION_()
	var_input
		idx: dint;
		ideal_idx: dint;
		capacity: dint;
	end
	{st}
	if (capacity <= 0) or (idx < 0) or (idx >= capacity) or (ideal_idx < 0) or (ideal_idx >= capacity) then
		eno := false;
		return;
	end_if;

	if idx >= ideal_idx then
		HashMap_private_static_robinhood_probedist := idx - ideal_idx;
	else
		HashMap_private_static_robinhood_probedist := idx + capacity - ideal_idx;
	end_if;
	{end}
end

(*{doc Find key slot by Robin Hood probing.
Stop early when probe-order guarantees no later match.
@inout table Entries.
@input key Key.
@input key_hash Hash.
@output slot Found slot or -1.
@output did_find Found flag.
@return true.
@error ENO is set to false on internal error. }*)
function HashMap_private_static_robinhood_find: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		table: array[*] of HashMapEntryType;
	end
	var_input
		key: HashMapKeyType;
		key_hash: udint;
	end
	var_output
		slot: dint;
		did_find: bool;
	end
	var_temp
		start_idx: dint;
		probe: dint;
		idx: dint;
		entry_ideal: dint;
		entry_probe: dint;
		is_eq: bool;
	end
	{st}
	start_idx := _HASHMAP_IDEAL_INDEX(key_hash);

	for probe := 0 to HashMapCapacity - 1 by 1 do
		idx := (start_idx + probe) mod HashMapCapacity;

		if table[idx].state = HASHMAP_SLOT_EMPTY then
			slot := -1;
			did_find := false;
			HashMap_private_static_robinhood_find := true;
			return;
		end_if;

		entry_ideal := _HASHMAP_IDEAL_INDEX(table[idx].hash);
		entry_probe := HashMap_private_static_robinhood_probedist(idx:=idx, ideal_idx:=entry_ideal, capacity:=HashMapCapacity, eno=>eno);
		if not eno then
			return;
		end_if;
		if entry_probe < probe then
			slot := -1;
			did_find := false;
			HashMap_private_static_robinhood_find := true;
			return;
		end_if;

		if table[idx].hash = key_hash then
			is_eq := HashMapKeyEqualsFunction(table[idx].key, key);
			if is_eq then
				slot := idx;
				did_find := true;
				HashMap_private_static_robinhood_find := true;
				return;
			end_if;
		end_if;
	end_for;

	slot := -1;
	did_find := false;
	HashMap_private_static_robinhood_find := true;
	{end}
end

(*{doc Initialize the container.
Reset entry states and count.
@inout self Target HashMap. }*)
function HashMap_init
	_PROLOGUE_OF_FUNCTION_()
	var_in_out self: HashMapCoreType; end
	var_temp
		idx: dint;
	end
	{st}
	self.count := 0;

	for idx := 0 to HashMapCapacity - 1 by 1 do
		self.entries[idx].state := HASHMAP_SLOT_EMPTY;
		// note: hash key
	end_for;

	{end}
end

(*{doc Copy container state.
Copy count and all table entries.
@inout self Source HashMap.
@inout dest Destination HashMap.
@return true on success. }*)
function HashMap_copy: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: HashMapCoreType;
		dest: HashMapCoreType;
	end
	var_temp
		idx: dint;
	end
	{st}
	dest.count := self.count;
	for idx := 0 to HashMapCapacity - 1 by 1 do
		dest.entries[idx] := self.entries[idx];
	end_for;

	HashMap_copy := true;
	{end}
end

(*{doc Insert or update a key/value pair.
Use Robin Hood insertion when key is not present.
@inout self Target HashMap.
@input key Key.
@input value Value.
@output replaced true when updated, false when inserted.
@return true on success, false on insert failure.
@error ENO is set to false on internal error. }*)
function HashMap_upsert: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out self: HashMapCoreType; end
	var_input
		key: HashMapKeyType;
		value: HashMapValueType;
	end
	var_output
		replaced: bool;
	end
	var_temp
		key_hash: udint;
		slot: dint;
		did_find: bool;
		max_count: dint;
		cur_hash: udint;
		cur_key: HashMapKeyType;
		cur_value: HashMapValueType;
		cur_start: dint;
		cur_probe: dint;
		idx: dint;
		entry_ideal: dint;
		entry_probe: dint;
		tmp_hash: udint;
		tmp_key: HashMapKeyType;
		tmp_value: HashMapValueType;
		scan_idx: dint;
	end
	{st}

	key_hash := HashMapKeyHashFunction(key);
	HashMap_private_static_robinhood_find(table:=self.entries, key:=key, key_hash:=key_hash, slot=>slot, did_find=>did_find, eno=>eno);
	if not eno then
		return;
	end_if;
	if did_find then
		self.entries[slot].value := value;
		replaced := true;
		HashMap_upsert := true;
		return;
	end_if;

	max_count := max((HashMapCapacity * HashMapMaxLoadPermille) / 1000, 1);
	if self.count >= max_count then
		replaced := false;
		HashMap_upsert := false;
		return;
	end_if;

	cur_hash := key_hash;
	cur_key := key;
	cur_value := value;
	cur_start := _HASHMAP_IDEAL_INDEX(cur_hash);
	cur_probe := 0;

	for scan_idx := 0 to HashMapCapacity - 1 by 1 do
		idx := (cur_start + cur_probe) mod HashMapCapacity;

		if self.entries[idx].state = HASHMAP_SLOT_EMPTY then
			self.entries[idx].state := HASHMAP_SLOT_USED;
			self.entries[idx].hash := cur_hash;
			self.entries[idx].key := cur_key;
			self.entries[idx].value := cur_value;
			self.count := self.count + 1;
			replaced := false;
			HashMap_upsert := true;
			return;
		end_if;

		entry_ideal := _HASHMAP_IDEAL_INDEX(self.entries[idx].hash);
		entry_probe := HashMap_private_static_robinhood_probedist(idx:=idx, ideal_idx:=entry_ideal, capacity:=HashMapCapacity, eno=>eno);
		if not eno then
			return;
		end_if;

		if entry_probe < cur_probe then
			tmp_hash := self.entries[idx].hash;
			tmp_key := self.entries[idx].key;
			tmp_value := self.entries[idx].value;

			self.entries[idx].hash := cur_hash;
			self.entries[idx].key := cur_key;
			self.entries[idx].value := cur_value;

			cur_hash := tmp_hash;
			cur_key := tmp_key;
			cur_value := tmp_value;
			cur_start := _HASHMAP_IDEAL_INDEX(cur_hash);
			cur_probe := entry_probe;
		end_if;

		cur_probe := cur_probe + 1;
	end_for;

	replaced := false;
	HashMap_upsert := false;
	{end}
end

(*{doc Lookup a value by key.
Read stored value for an existing key.
@inout self Target HashMap.
@input key Target key.
@inout value Output value.
@error ENO is set to false when the key is not found or on internal error. }*)
function HashMap_lookup
	_PROLOGUE_OF_FUNCTION_()
	var_in_out self: HashMapCoreType; end
	var_input
		key: HashMapKeyType;
	end
	var_in_out
		value: HashMapValueType;
	end
	var_temp
		key_hash: udint;
		slot: dint;
		did_find: bool;
	end
	{st}
	key_hash := HashMapKeyHashFunction(key);

	HashMap_private_static_robinhood_find(table:=self.entries, key:=key, key_hash:=key_hash, slot=>slot, did_find=>did_find, eno=>eno);
	if not eno then
		return;
	end_if;
	if not did_find then
		eno := false;
		return;
	end_if;

	value := self.entries[slot].value;
	{end}
end

(*{doc Delete a key/value pair.
Apply backward-shift compaction after removal.
@inout self Target HashMap.
@input key Target key to delete.
@return true when deleted, false when the key is not found.
@error ENO is set to false on internal error. }*)
function HashMap_delete: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out self: HashMapCoreType; end
	var_input
		key: HashMapKeyType;
	end
	var_temp
		key_hash: udint;
		slot: dint;
		did_find: bool;
		shift_idx, scan_idx: dint;
		next_ideal: dint;
		next_probe: dint;
	end
	{st}
	key_hash := HashMapKeyHashFunction(key);
	HashMap_private_static_robinhood_find(table:=self.entries, key:=key, key_hash:=key_hash, slot=>slot, did_find=>did_find, eno=>eno);
	if not eno then
		return;
	end_if;
	if not did_find then
		HashMap_delete := false;
		return;
	end_if;

	shift_idx := slot;
	scan_idx := (shift_idx + 1) mod HashMapCapacity;
	while self.entries[scan_idx].state = HASHMAP_SLOT_USED do
		next_ideal := _HASHMAP_IDEAL_INDEX(self.entries[scan_idx].hash);
		next_probe := HashMap_private_static_robinhood_probedist(idx:=scan_idx, ideal_idx:=next_ideal, capacity:=HashMapCapacity, eno=>eno);
		if not eno then
			return;
		end_if;
		if next_probe = 0 then
			exit;
		end_if;

		self.entries[shift_idx] := self.entries[scan_idx];
		shift_idx := scan_idx;
		scan_idx := (scan_idx + 1) mod HashMapCapacity;
	end_while;

	self.entries[shift_idx].state := HASHMAP_SLOT_EMPTY;
	// note: hash key
	self.count := self.count - 1;

	HashMap_delete := true;
	{end}
end

(*{doc Copy all keys into an output array.
Export keys from used slots in table order.
@inout self Target HashMap.
@inout keys Output array.
@output written Number of copied keys.
@return true on success.
@error ENO is set to false when output capacity is insufficient. }*)
function HashMap_keys: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out self: HashMapCoreType; end
	var_in_out
		keys: array[*] of HashMapKeyType;
	end
	var_output
		written: dint;
	end
	var_temp
		idx: dint;
		out_idx: dint;
		out_hi: dint;
	end
	{st}
	out_idx := lower_bound(keys, 1);
	out_hi := upper_bound(keys, 1);
	written := 0;

	for idx := 0 to HashMapCapacity - 1 by 1 do
		if self.entries[idx].state = HASHMAP_SLOT_USED then
			if out_idx > out_hi then
				eno := false;
				return;
			end_if;
			keys[out_idx] := self.entries[idx].key;
			out_idx := out_idx + 1;
			written := written + 1;
		end_if;
	end_for;

	HashMap_keys := true;
	{end}
end

(*{doc Compare two HashMaps for equality.
Require equal counts and equal values for each key.
@inout self First HashMap.
@inout other Second HashMap.
@return true when all keys and values match. }*)
function HashMap_equals: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: HashMapCoreType;
		other: HashMapCoreType;
	end
	var_temp
		keys: array[HashMapCapacity] of HashMapKeyType;
		written: dint;
		count_self: dint;
		count_other: dint;
		idx: dint;
		v_self: HashMapValueType;
		v_other: HashMapValueType;
	end
	{st}
	count_self := HashMap_len(self:=self);
	count_other := HashMap_len(self:=other);

	if count_self <> count_other then
		HashMap_equals := false;
		return;
	end_if;

	HashMap_keys(self:=self, keys:=keys, written=>written, eno=>eno);
	if not eno then
		return;
	end_if;

	for idx := 0 to written - 1 by 1 do
		HashMap_lookup(self:=self, key:=keys[idx], value:=v_self, eno=>eno);
		if not eno then
			return;
		end_if;

		HashMap_lookup(self:=other, key:=keys[idx], value:=v_other, eno=>eno);
		if not eno then
			return;
		end_if;
		if not _HASHMAP_VALUE_EQUALS_(v_self, v_other) then
			HashMap_equals := false;
			return;
		end_if;
	end_for;

	HashMap_equals := true;
	{end}
end

{#ifdef HashMapValueHashFunction}
(*{doc Compute a hash value from all key/value pairs.
Combine key and value hashes with order-insensitive accumulation.
@inout self Target HashMap.
@return Hash value. }*)
function HashMap_hash: udint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: HashMapCoreType;
	end
	var_temp
		keys: array[HashMapCapacity] of HashMapKeyType;
		written: dint;
		idx: dint;
		key_hash, value_hash: udint;
		value: HashMapValueType;
		acc1, acc2: udint;
		cnt: dint;
	end
	{st}
	acc1 := udint#2166136261;
	acc2 := udint#0;

	HashMap_keys(self:=self, keys:=keys, written=>written, eno=>eno);
	if not eno then
		return;
	end_if;

	for idx := 0 to written - 1 by 1 do
		key_hash := HashMapKeyHashFunction(keys[idx]);
		HashMap_lookup(self:=self, key:=keys[idx], value:=value, eno=>eno);
		if not eno then
			return;
		end_if;
		value_hash := HashMapValueHashFunction(value);
		acc1 := acc1 + dword_to_udint(udint_to_dword(key_hash) xor udint_to_dword(value_hash * udint#16777619 + udint#31));
		acc2 := acc2 + (key_hash + udint#7) * (value_hash + udint#13);
	end_for;

	cnt := HashMap_len(self:=self);

	HashMap_hash := acc1 * udint#2246822519 + acc2 + dint_to_udint(cnt);
	{end}
end
{#endif}

(*{doc Return true when the container is empty.
Check whether count is zero.
@inout self Target HashMap.
@return true if entry count is 0, otherwise false. }*)
function HashMap_empty: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out self: HashMapCoreType; end
	{st}
	HashMap_empty := self.count = 0;
	{end}
end

(*{doc Clear all entries.
Mark all slots empty and reset count.
@inout self Target HashMap.
@return true on success. }*)
function HashMap_clear: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out self: HashMapCoreType; end
	var_temp
		idx: dint;
	end
	{st}
	for idx := 0 to HashMapCapacity - 1 by 1 do
		self.entries[idx].state := HASHMAP_SLOT_EMPTY;
		// note: hash key
	end_for;
	self.count := 0;

	HashMap_clear := true;
	{end}
end
