(*
	================================================================================
	Dequeue Template Generator - Macro-based Generic Dequeue
	================================================================================

	This template generates a typed dequeue from preprocessor macros.

	Usage:
	1. Define required macros.
	2. Optionally define behavior macros.
	3. Include this template once per generated type.

	Required macros:
	  - DequeueName
	    Suffix used for generated types and functions.
	  - DequeueItemType
	    Element type stored in the dequeue.
	  - DequeueCapacity
	    Maximum number of elements.

	Optional macros:
	  - DequeueElementEqualsFunction
	    Equality function for contains/equals.
	    Default: direct value comparison.
	  - DequeueItemHashFunction
	    Hash function for hash API generation.

	Generated APIs:
	  - init, copy, append, appendleft, pop, popleft
	  - first, last, contains, elements, len, clear, empty, equals
	  - hash (when DequeueItemHashFunction is defined)

	Example (dint):
	  {#define DequeueName dint}
	  {#define DequeueItemType dint}
	  {#define DequeueCapacity 256}
	  {#include <container/dequeue/dequeue_template.txt>}

	Example (string):
	  {#define DequeueName string_ci}
	  {#define DequeueItemType string[128]}
	  {#define DequeueCapacity 64}
	  {#define DequeueElementEqualsFunction String_equals}
	  {#include <container/dequeue/dequeue_template.txt>}

	Algorithm:
	  Ring buffer with O(1) amortized push/pop on both ends.
	================================================================================
*)

type
	(*{doc Core state for a typed dequeue instance.}*)
	DequeueCoreType: struct
		(*{doc Backing storage.}*)
		items: array[DequeueCapacity] of DequeueItemType;
		(*{doc Logical index of the first element.}*)
		begin_idx: dint;
		(*{doc Logical index one past the last element.}*)
		end_idx: dint;
		(*{doc Current number of elements.}*)
		count: dint;
		(*{doc Maximum number of elements.}*)
		capacity: dint;
	end
end

(*{doc Initialize dequeue state.
Set indices and counters to the empty state.
@inout self Target Dequeue. }*)
function Dequeue_init
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: DequeueCoreType;
	end
	{st}
	self.begin_idx := 0;
	self.end_idx := 0;
	self.count := 0;
	self.capacity := DequeueCapacity;
	{end}
end

(*{doc Copy dequeue state.
Copy all fields and backing storage.
@inout self Source dequeue.
@inout dest Destination dequeue.
@return true on success. }*)
function Dequeue_copy: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: DequeueCoreType;
		dest: DequeueCoreType;
	end
	var_temp
		idx: dint;
	end
	{st}
	dest.begin_idx := self.begin_idx;
	dest.end_idx := self.end_idx;
	dest.count := self.count;
	dest.capacity := self.capacity;

	for idx := 0 to DequeueCapacity - 1 by 1 do
		dest.items[idx] := self.items[idx];
	end_for;

	Dequeue_copy := true;
	{end}
end

(*{doc Append one element at the back.
Insert one element at the logical tail position.
@inout self Target Dequeue.
@input item Element to append.
@return true on success, false when full or invalid capacity.
@error ENO is set to false on error. }*)
function Dequeue_append: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: DequeueCoreType;
	end
	var_input
		item: DequeueItemType;
	end
	var_temp
		slot: dint;
	end
	{st}
	if (self.capacity <= 0) or (self.count = self.capacity) then
		Dequeue_append := false;
		return;
	end_if;

	slot := self.end_idx mod self.capacity;
	if slot < 0 then
		slot := slot + self.capacity;
	end_if;
	self.items[slot] := item;
	self.end_idx := (slot + 1) mod self.capacity;
	self.count := self.count + 1;

	Dequeue_append := true;
	{end}
end

(*{doc Append one element at the front.
Insert one element before the current front position.
@inout self Target Dequeue.
@input item Element to append.
@return true on success, false when full or invalid capacity.
@error ENO is set to false on error. }*)
function Dequeue_appendleft: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: DequeueCoreType;
	end
	var_input
		item: DequeueItemType;
	end
	var_temp
		slot: dint;
	end
	{st}
	if (self.capacity <= 0) or (self.count = self.capacity) then
		Dequeue_appendleft := false;
		return;
	end_if;

	slot := (self.begin_idx - 1) mod self.capacity;
	if slot < 0 then
		slot := slot + self.capacity;
	end_if;

	self.items[slot] := item;
	self.begin_idx := slot;
	self.count := self.count + 1;

	Dequeue_appendleft := true;
	{end}
end

(*{doc Remove one element from the back.
Return the logical tail element and shrink size by one.
@inout self Target Dequeue.
@output item Removed element.
@return true on success, false when empty or invalid capacity.
@error ENO is set to false on error. }*)
function Dequeue_pop: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: DequeueCoreType;
	end
	var_output
		item: DequeueItemType;
	end
	var_temp
		slot: dint;
	end
	{st}
	if (self.capacity <= 0) or (self.count = 0) then
		Dequeue_pop := false;
		return;
	end_if;

	slot := (self.end_idx - 1) mod self.capacity;
	if slot < 0 then
		slot := slot + self.capacity;
	end_if;

	item := self.items[slot];
	self.end_idx := slot;
	self.count := self.count - 1;

	Dequeue_pop := true;
	{end}
end

(*{doc Check whether an element exists.
Scan elements in dequeue order and compare by configured equality.
@inout self Target Dequeue.
@input item Element to search.
@return true when found, otherwise false.
@error ENO is set to false on error. }*)
function Dequeue_contains: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: DequeueCoreType;
	end
	var_input
		item: DequeueItemType;
	end
	var_temp
		idx: dint;
		slot: dint;
	end
	{st}
	if (self.capacity <= 0) or (self.count = 0) then
		Dequeue_contains := false;
		return;
	end_if;

	for idx := 0 to self.count - 1 by 1 do
		slot := (self.begin_idx + idx) mod self.capacity;
		if slot < 0 then
			slot := slot + self.capacity;
		end_if;
		if _DEQUEUE_ELEM_EQUALS_(self.items[slot], item) then
			Dequeue_contains := true;
			return;
		end_if;
	end_for;

	Dequeue_contains := false;
	{end}
end

(*{doc Export elements in dequeue order.
Write elements from front to back into the destination array.
@inout self Source dequeue.
@inout elements Destination array.
@output written Number of exported elements.
@return true on success.
@error ENO is set to false when destination capacity is insufficient. }*)
function Dequeue_elements: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: DequeueCoreType;
	end
	var_in_out
		elements: array[*] of DequeueItemType;
	end
	var_output
		written: dint;
	end
	var_temp
		idx: dint;
		slot: dint;
		out_idx: dint;
		out_hi: dint;
	end
	{st}
	out_idx := lower_bound(elements, 1);
	out_hi := upper_bound(elements, 1);
	written := 0;

	for idx := 0 to self.count - 1 by 1 do
		if out_idx > out_hi then
			eno := false;
			return;
		end_if;

		slot := (self.begin_idx + idx) mod self.capacity;
		if slot < 0 then
			slot := slot + self.capacity;
		end_if;

		elements[out_idx] := self.items[slot];
		out_idx := out_idx + 1;
		written := written + 1;
	end_for;

	Dequeue_elements := true;
	{end}
end

(*{doc Compare two dequeues.
Compare size and each element in dequeue order.
@inout self Left dequeue.
@inout other Right dequeue.
@return true when length and order are equal. }*)
function Dequeue_equals: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: DequeueCoreType;
		other: DequeueCoreType;
	end
	var_temp
		idx: dint;
		self_slot: dint;
		other_slot: dint;
		v_self: DequeueItemType;
		v_other: DequeueItemType;
	end
	{st}
	if self.count <> other.count then
		Dequeue_equals := false;
		return;
	end_if;

	for idx := 0 to self.count - 1 by 1 do
		self_slot := (self.begin_idx + idx) mod self.capacity;
		if self_slot < 0 then
			self_slot := self_slot + self.capacity;
		end_if;

		other_slot := (other.begin_idx + idx) mod other.capacity;
		if other_slot < 0 then
			other_slot := other_slot + other.capacity;
		end_if;

		v_self := self.items[self_slot];
		v_other := other.items[other_slot];
		if not _DEQUEUE_ELEM_EQUALS_(v_self, v_other) then
			Dequeue_equals := false;
			return;
		end_if;
	end_for;

	Dequeue_equals := true;
	{end}
end

{#ifdef DequeueItemHashFunction}
(*{doc Compute a hash in dequeue order.
Combine item hashes with order-sensitive accumulation.
@inout self Target Dequeue.
@return Hash value. }*)
function Dequeue_hash: udint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		self: DequeueCoreType;
	end
	var_temp
		idx: dint;
		slot: dint;
		item_hash: udint;
		acc: udint;
	end
	{st}
	acc := udint#2166136261;
	for idx := 0 to self.count - 1 by 1 do
		slot := (self.begin_idx + idx) mod self.capacity;
		if slot < 0 then
			slot := slot + self.capacity;
		end_if;
		item_hash := DequeueItemHashFunction(self.items[slot]);
		acc := dword_to_udint(udint_to_dword(acc) xor udint_to_dword(item_hash)) * udint#16777619 + dint_to_udint(idx + 1);
	end_for;

	Dequeue_hash := acc + dint_to_udint(self.count);
	{end}
end
