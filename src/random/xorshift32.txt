(*{doc This file contains the implementation of the Xorshift32 algorithm.
Xorshift32 is a simple and fast pseudorandom number generator.
It is based on the XOR and bit-shifting operations. }*)

type
	Xorshift32_t: struct
		state: dword;
	end
end

(*{doc Initializes the Xorshift32 pseudorandom number generator.
@inout self The Xorshift32 generator state.
@input seed The initial seed value. It must be a 32-bit bits value and non-zero. }*)
function Xorshift32_init
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The Xorshift32 generator state.}*)
		self: Xorshift32_t;
	end
	var_input
		(*{doc The initial seed value. It must be a 32-bit bits value and non-zero.}*)
		seed: dword;
	end
	{st}
	self.state := seed;
	{end}
end

(*{doc Generates a pseudo-random number.
@inout self The Xorshift32 generator state.
@input bits The number of significant bits (1..32). Default: 32.
@return A pseudo-random number. }*)
function Xorshift32_next: dword
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The Xorshift32 generator state.}*)
		self: Xorshift32_t;
	end
	var_input
		(*{doc The number of significant bits (1..32). Default: 32.}*)
		bits: dint := 32;
	end
	var_temp
		xs: dword;
	end
	{st}
	xs := self.state;
	xs := xs xor shl(xs, 13);
	xs := xs xor shr(xs, 17);
	xs := xs xor shl(xs, 5);
	self.state := xs;
	Xorshift32_next := shr(xs, 32 - bits);
	{end}
end

(*{doc Generates a uniformly distributed boolean pseudo-random number.
@inout self The Xorshift32 generator state.
@return true or false. }*)
function Xorshift32_nextBool: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The Xorshift32 generator state.}*)
		self: Xorshift32_t;
	end
	{st}
	Xorshift32_nextBool := Xorshift32_next(self, 1) = dword#1;
	{end}
end

(*{doc Generates a uniformly distributed unsigned 32-bit integer pseudo-random number.
@inout self The Xorshift32 generator state.
@return An unsigned 32-bit integer pseudo-random number, between 0 and 4294967295. }*)
function Xorshift32_nextUint32: udint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The Xorshift32 generator state.}*)
		self: Xorshift32_t;
	end
	{st}
	Xorshift32_nextUint32 := dword_to_udint(Xorshift32_next(self, 32));
	{end}
end
{#define Xorshift32_nextUDINT Xorshift32_nextUint32} (* alias *)

{#if not (_KEYENCE)}
(*{doc Generates a uniformly distributed unsigned 64-bit integer pseudo-random number.
@inout self The Xorshift32 generator state.
@return An unsigned 64-bit integer pseudo-random number, between 0 and 18446744073709551615. }*)
function Xorshift32_nextUint64: ulint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The Xorshift32 generator state.}*)
		self: Xorshift32_t;
	end
	var_temp
		hb, lb: lword;
	end
	{st}
	hb := Xorshift32_next(self, 32);
	lb := Xorshift32_next(self, 32);
	Xorshift32_nextUint64 := lword_to_ulint(shl(hb, 32) or lb);
	{end}
end
{#define Xorshift32_nextULINT Xorshift32_nextUint64} (* alias *)
{#endif}

(*{doc Generates a 32-bit floating-point pseudo-random number with uniform distribution in the range 0.0 to 1.0.
@inout self The Xorshift32 generator state.
@return [0.0, 1.0) of the half-open interval. }*)
function Xorshift32_nextFloat: real
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The Xorshift32 generator state.}*)
		self: Xorshift32_t;
	end
	{st}
	Xorshift32_nextFloat := lreal_to_real(
		udint_to_lreal(dword_to_udint(Xorshift32_next(self, 32))) / lreal#4294967296.0); // 2^32
	{end}
end
{#define Xorshift32_nextREAL Xorshift32_nextFloat} (* alias *)

(*{doc Generates a 53-bit floating-point pseudo-random number with uniform distribution in the range 0.0 to 1.0.
@inout self The Xorshift32 generator state.
@return [0.0, 1.0) of the half-open interval. }*)
function Xorshift32_nextDouble: lreal
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The Xorshift32 generator state.}*)
		self: Xorshift32_t;
	end
	var_temp
		av, bv: lreal;
	end
	{st}
	av := udint_to_lreal(dword_to_udint(Xorshift32_next(self, 32 - 5)));
	bv := udint_to_lreal(dword_to_udint(Xorshift32_next(self, 32 - 6)));
	Xorshift32_nextDouble := (av * lreal#67108864.0 + bv) / lreal#9007199254740992.0;
	{end}
end
{#define Xorshift32_nextLREAL Xorshift32_nextDouble} (* alias *)
