(*{doc This file contains the implementation of the LCG (Linear Congruential Generator) algorithm.
LCG is an ultra-lightweight deterministic PRNG suitable for resource-constrained embedded PLCs.
Formula: state := 1664525 * state + 1013904223 (mod 2^32)
Parameters from Numerical Recipes (widely used standard parameters). }*)

type
	LCG_t: struct
		state: udint;
	end
end

(*{doc Initialize the LCG PRNG.
@inout self The LCG generator state.
@input seed The initial seed value. }*)
function LCG_init
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The LCG generator state.}*)
		self: LCG_t;
	end
	var_input
		(*{doc The initial seed value.}*)
		seed: udint;
	end
	{st}
	self.state := seed;
	{end}
end

(*{doc Generates an unsigned 32-bit pseudo-random number.
@inout self The LCG generator state.
@return An unsigned 32-bit pseudo-random number, between 0 and 4294967295. }*)
function LCG_nextUint32: udint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The LCG generator state.}*)
		self: LCG_t;
	end
	{st}
	self.state := udint#1664525 * self.state + udint#1013904223;
	LCG_nextUint32 := self.state;
	{end}
end
{#define LCG_nextUDINT LCG_nextUint32} (* alias *)

(*{doc Generates a uniformly distributed boolean pseudo-random number using the MSB.
@inout self The LCG generator state.
@return true or false. }*)
function LCG_nextBool: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The LCG generator state.}*)
		self: LCG_t;
	end
	{st}
	LCG_nextBool := shr(udint_to_dword(LCG_nextUint32(self)), 31) = dword#1;
	{end}
end

(*{doc Generates a 32-bit floating-point pseudo-random number with uniform distribution in the range 0.0 to 1.0.
@inout self The LCG generator state.
@return [0.0, 1.0) of the half-open interval. }*)
function LCG_nextFloat: real
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The LCG generator state.}*)
		self: LCG_t;
	end
	{st}
	LCG_nextFloat := lreal_to_real(
		udint_to_lreal(LCG_nextUint32(self)) / lreal#4294967296.0); // 2^32
	{end}
end
{#define LCG_nextREAL LCG_nextFloat} (* alias *)

(*{doc Generates a 53-bit floating-point pseudo-random number with uniform distribution in the range 0.0 to 1.0.
@inout self The LCG generator state.
@return [0.0, 1.0) of the half-open interval. }*)
function LCG_nextDouble: lreal
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The LCG generator state.}*)
		self: LCG_t;
	end
	var_temp
		av, bv: lreal;
	end
	{st}
	av := udint_to_lreal(dword_to_udint(shr(udint_to_dword(LCG_nextUint32(self)), 5)));  // upper 27 bits
	bv := udint_to_lreal(dword_to_udint(shr(udint_to_dword(LCG_nextUint32(self)), 6)));  // upper 26 bits
	LCG_nextDouble := (av * lreal#67108864.0 + bv) / lreal#9007199254740992.0;
	{end}
end
{#define LCG_nextLREAL LCG_nextDouble} (* alias *)
