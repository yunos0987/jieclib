{#define N 624}
{#define M 397}
{#define N_1 623} // for a omron bug.
{#define M_1 396} // for a omron bug.
{#define MATRIX_A dword#16#9908b0df}   (* constant vector a *)
{#define UPPER_MASK dword#16#80000000} (* most significant w-r bits *)
{#define LOWER_MASK dword#16#7fffffff} (* least significant r bits *)

(*{doc Mersenne Twister POUs.
The Mersenne Twister, a method for generating pseudo-random numbers, is implemented here.
This implementation generates uniform random numbers of either unsigned integer type
or floating-point type using the Mersenne Twister algorithm.
It initializes random number generation with a single integer,
but can also accommodate arbitrary bit-width values using an array if necessary.

These POUs are the IEC 61131-3 ST language port from the following site (C language source).
http://www.math.sci.hiroshima-u.ac.jp/m-mat/MT/MT2002/CODES/mt19937ar.c
The external interface is modeled after the java.util.Random class of Java(TM) language.
https://docs.oracle.com/javase/jp/8/docs/api/java/util/Random.html

e.g.
// Create an instance and initialize the random number generator with a bits value.
var mt: MersenneTwister_t; end_var
MersenneTwister_init(mt, dword#16#cafe_babe);
// Uniform random number of 32-bit unsigned integer type.
random_num := MersenneTwister_nextUint32(mt); }*)

type
	MersenneTwister_t: struct
		mt: array[N] of dword;
		mti: dint;
	end_struct;
end_type

(*{doc Initialize the MersenneTwister PRNG with a single seed value.
@inout self The MersenneTwister generator state.
@input seed The initial seed value. }*)
function MersenneTwister_init
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The MersenneTwister generator state.}*)
		self: MersenneTwister_t;
	end_var
	var_input
		(*{doc The initial seed value.}*)
		seed: dword;
	end_var
	var_temp
		idx: dint;
	end_var
	// {st}
	self.mt[0] := seed;
	for idx := 1 to N - 1 by 1 do
		self.mt[idx] := udint_to_dword(udint#1812433253 * dword_to_udint(self.mt[idx - 1] xor shr(self.mt[idx - 1], 30)) + dint_to_udint(idx));
	end_for;
	self.mti := N;
	// {end}
end_function

(*{doc Initialize the MersenneTwister PRNG with an array of seed values.
@inout self The MersenneTwister generator state.
@inout seeds The seed array.
@input seeds_length The length of the seeds array. }*)
function MersenneTwister_initByArray
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The MersenneTwister generator state.}*)
		self: MersenneTwister_t;
	end_var
	var_in_out
		(*{doc The seed array.}*)
		seeds: array[*] of dword;
	end_var
	var_input
		(*{doc The length of the seeds array.}*)
		seeds_length: dint;
	end_var
	var_temp
		mt_idx, seed_idx, loop_cnt: dint;
	end_var
	// {st}
	MersenneTwister_init(self, dword#19650218);
	mt_idx := 1;
	seed_idx := 0;
	if N > seeds_length then
		loop_cnt := N;
	else
		loop_cnt := seeds_length;
	end_if;
	for loop_cnt := loop_cnt to 1 by -1 do
		self.mt[mt_idx] := udint_to_dword(
			dword_to_udint(self.mt[mt_idx] xor udint_to_dword(dword_to_udint(self.mt[mt_idx - 1] xor shr(self.mt[mt_idx - 1], 30)) * udint#1664525))
			+ dword_to_udint(seeds[seed_idx]) + dint_to_udint(seed_idx));
		mt_idx := mt_idx + 1;
		seed_idx := seed_idx + 1;
		if mt_idx >= N then
			self.mt[0] := self.mt[N_1];
			mt_idx := 1;
		end_if;
		if seed_idx >= seeds_length then
			seed_idx := 0;
		end_if;
	end_for;
	for loop_cnt := N - 1 to 1 by -1 do
		self.mt[mt_idx] := udint_to_dword(
			dword_to_udint(self.mt[mt_idx] xor udint_to_dword(dword_to_udint(self.mt[mt_idx - 1] xor shr(self.mt[mt_idx - 1], 30)) * udint#1566083941))
			- dint_to_udint(mt_idx));
		mt_idx := mt_idx + 1;
		if mt_idx >= N then
			self.mt[0] := self.mt[N_1];
			mt_idx := 1;
		end_if;
	end_for;

	self.mt[0] := dword#16#80000000;
	// {end}
end_function

(*{doc Generates a pseudo-random number.
@inout self The MersenneTwister generator state.
@input bits The number of significant bits (1..32). Default: 32.
@return A pseudo-random number. }*)
function MersenneTwister_next: dword
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The MersenneTwister generator state.}*)
		self: MersenneTwister_t;
	end_var
	var_input
		(*{doc The number of significant bits (1..32). Default: 32.}*)
		bits: dint := 32;
	end_var
	var_temp
		yw: dword;
		kk: dint;
	end_var
	var_temp constant
		mag01: array[0..1] of dword := [dword#16#0, MATRIX_A];
	end_var
	// {st}
	if self.mti >= N then
		if self.mti = 625 then
			MersenneTwister_init(self, dword#5489);
		end_if;

		for kk := 0 to (N - 1) - M by 1 do
			yw := (self.mt[kk] and UPPER_MASK) or (self.mt[kk + 1] and LOWER_MASK);
			self.mt[kk] := self.mt[kk + M] xor shr(yw, 1) xor mag01[dword_to_dint(yw and dword#16#1)];
		end_for;
		for kk := kk to (N - 1) - 1 by 1 do
			yw := (self.mt[kk] and UPPER_MASK) or (self.mt[kk + 1] and LOWER_MASK);
			self.mt[kk] := self.mt[kk + M - N] xor shr(yw, 1) xor mag01[dword_to_dint(yw and dword#16#1)];
		end_for;
		yw := (self.mt[N_1] and UPPER_MASK) or (self.mt[0] and LOWER_MASK);
		self.mt[N_1] := self.mt[M_1] xor shr(yw, 1) xor mag01[dword_to_dint(yw and dword#16#1)];
		self.mti := 0;
	end_if;

	yw := self.mt[self.mti];
	self.mti := self.mti + 1;

	yw := yw xor shr(yw, 11);
	yw := yw xor (shl(yw,  7) and dword#16#9d2c5680);
	yw := yw xor (shl(yw, 15) and dword#16#efc60000);
	yw := yw xor shr(yw, 18);

	MersenneTwister_next := shr(yw, 32 - bits);
	// {end}
end_function

(*{doc Generates a uniformly distributed boolean pseudo-random number.
@inout self The MersenneTwister generator state.
@return true or false. }*)
function MersenneTwister_nextBool: bool
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The MersenneTwister generator state.}*)
		self: MersenneTwister_t;
	end_var
	// {st}
	MersenneTwister_nextBool := MersenneTwister_next(self, 1) = dword#1;
	// {end}
end_function

(*{doc Generates a uniformly distributed unsigned 32-bit integer pseudo-random number.
@inout self The MersenneTwister generator state.
@return An unsigned 32-bit integer pseudo-random number, between 0 and 4294967295. }*)
function MersenneTwister_nextUint32: udint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The MersenneTwister generator state.}*)
		self: MersenneTwister_t;
	end_var
	// {st}
	MersenneTwister_nextUint32 := dword_to_udint(MersenneTwister_next(self, 32));
	// {end}
end_function
{#define MersenneTwister_nextUDINT MersenneTwister_nextUint32} (* alias *)

{#if not (_KEYENCE)}
(*{doc Generates a uniformly distributed unsigned 64-bit integer pseudo-random number.
@inout self The MersenneTwister generator state.
@return An unsigned 64-bit integer pseudo-random number, between 0 and 18446744073709551615. }*)
function MersenneTwister_nextUint64: ulint
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The MersenneTwister generator state.}*)
		self: MersenneTwister_t;
	end_var
	var_temp
		hi_word, lo_word: lword;
	end_var
	// {st}
	hi_word := MersenneTwister_next(self, 32);
	lo_word := MersenneTwister_next(self, 32);
	MersenneTwister_nextUint64 := lword_to_ulint(shl(hi_word, 32) or lo_word);
	// {end}
end_function
{#define MersenneTwister_nextULINT MersenneTwister_nextUint64} (* alias *)
{#endif}

(*{doc Generates a 32-bit floating-point pseudo-random number with uniform distribution in the range 0.0 to 1.0.
@inout self The MersenneTwister generator state.
@return [0.0, 1.0) of the half-open interval. }*)
function MersenneTwister_nextFloat: real
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The MersenneTwister generator state.}*)
		self: MersenneTwister_t;
	end_var
	// {st}
	MersenneTwister_nextFloat := lreal_to_real(
		udint_to_lreal(dword_to_udint(MersenneTwister_next(self, 32))) / lreal#4294967296.0); // 2^32
	// {end}
end_function
{#define MersenneTwister_nextREAL MersenneTwister_nextFloat} (* alias *)

(*{doc Generates a 53-bit floating-point pseudo-random number with uniform distribution in the range 0.0 to 1.0.
@inout self The MersenneTwister generator state.
@return [0.0, 1.0) of the half-open interval. }*)
function MersenneTwister_nextDouble: lreal
	_PROLOGUE_OF_FUNCTION_()
	var_in_out
		(*{doc The MersenneTwister generator state.}*)
		self: MersenneTwister_t;
	end_var
	var_temp
		hi_bits, lo_bits: lreal;
	end_var
	// {st}
	hi_bits := udint_to_lreal(dword_to_udint(MersenneTwister_next(self, 32 - 5)));
	lo_bits := udint_to_lreal(dword_to_udint(MersenneTwister_next(self, 32 - 6)));
	MersenneTwister_nextDouble := (hi_bits * lreal#67108864.0 + lo_bits) / lreal#9007199254740992.0;
	// {end}
end_function
{#define MersenneTwister_nextLREAL MersenneTwister_nextDouble} (* alias *)
