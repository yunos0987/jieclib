	_JIEC_TESTNAMES_LIST_(

	)
)

TEST_S(test_array_equals)
	var
		aa: array[0..3] of lreal := [1.0, 2.0, 3.0, 4.0];
		bb: array[0..3] of lreal := [1.0, 2.0, 3.0, 4.0];
		cc: array[0..3] of lreal := [1.0, 2.0, 0.0, 4.0];
		pp: array[0..3] of dint  := [1, 2, 3, 4];
		qq: array[0..3] of dint  := [1, 2, 3, 4];
		rr: array[0..3] of dint  := [1, 2, 3, 9];
		aa3: array[0..2] of lreal := [1.0, 2.0, 3.0];
		ok: bool;
	end
	{st}
	// Fact: Two lreal arrays with equal elements compare as equal.
		// Arrange
		// Act
	ok := Array_equals_lreal(self:=aa, other:=bb);
		// Assert
	EXPECT_TRUE(ok);

	// Fact: Two lreal arrays differing in one element compare as not equal.
		// Arrange
		// Act
	ok := Array_equals_lreal(self:=aa, other:=cc);
		// Assert
	EXPECT_FALSE(ok);

	// Fact: Two dint arrays with equal elements compare as equal.
		// Arrange
		// Act
	ok := Array_equals_dint(self:=pp, other:=qq);
		// Assert
	EXPECT_TRUE(ok);

	// Fact: Two dint arrays differing in the last element compare as not equal.
		// Arrange
		// Act
	ok := Array_equals_dint(self:=pp, other:=rr);
		// Assert
	EXPECT_FALSE(ok);

	// Fact: Arrays of different lengths compare as not equal.
		// Arrange
		// Act
	ok := Array_equals_lreal(self:=aa, other:=aa3);
		// Assert
	EXPECT_FALSE(ok);
	{end}
END_TEST_S
