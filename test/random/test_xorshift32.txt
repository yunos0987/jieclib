TEST_S(test_xorshift32_nextIntInRange)
	var
		prng: Xorshift32_t;
		v: dint;
		ok: bool;
	end
	{st}
	// min > max → error
	Xorshift32_init(prng, dword#16#92d68ca2);
	Xorshift32_nextIntInRange(self:=prng, min_val:=10, max_val:=5, eno=>ok);
	EXPECT_FALSE(ok);
	// min == max → always returns min
	Xorshift32_init(prng, dword#16#92d68ca2);
	EXPECT_EQ_DINT(dint#3, Xorshift32_nextIntInRange(prng, 3, 3));
	// result must be in [min_val, max_val]
	Xorshift32_init(prng, dword#16#92d68ca2);
	v := Xorshift32_nextIntInRange(prng, -100, 100);
	EXPECT_GE_DINT(v, -100);
	EXPECT_LE_DINT(v, 100);
	// Python reference value (seed=0x92d68ca2, range=-100..100)
	EXPECT_EQ_DINT(dint#54, v);
	{end}
END_TEST_S

TEST_S(test_xorshift32_nextRealInRange)
	var
		prng: Xorshift32_t;
		v: lreal;
		ok: bool;
	end
	{st}
	// min >= max → error
	Xorshift32_init(prng, dword#16#92d68ca2);
	Xorshift32_nextRealInRange(self:=prng, min_val:=lreal#5.0, max_val:=lreal#5.0, eno=>ok);
	EXPECT_FALSE(ok);
	// result must be in [min_val, max_val)
	Xorshift32_init(prng, dword#16#92d68ca2);
	v := Xorshift32_nextRealInRange(prng, lreal#-1.0, lreal#1.0);
	EXPECT_GE_LREAL(v, lreal#-1.0);
	EXPECT_LT_LREAL(v, lreal#1.0);
	// Python reference value (seed=0x92d68ca2, range=-1.0..1.0)
	EXPECT_NEAR(lreal#-6.631072225948642e-01, v, lreal#1.0e-15);
	{end}
END_TEST_S

TEST_S(test_xorshift32_nextBernoulli)
	var
		prng: Xorshift32_t;
	end
	{st}
	// prob=1.0: always true
	Xorshift32_init(prng, dword#16#92d68ca2);
	EXPECT_EQ_BOOL(true, Xorshift32_nextBernoulli(prng, lreal#1.0));
	// prob=0.0: always false
	Xorshift32_init(prng, dword#16#92d68ca2);
	EXPECT_EQ_BOOL(false, Xorshift32_nextBernoulli(prng, lreal#0.0));
	// Python reference value: seed=0x92d68ca2, nextDouble=0.168... < 0.2 → true
	Xorshift32_init(prng, dword#16#92d68ca2);
	EXPECT_EQ_BOOL(true, Xorshift32_nextBernoulli(prng, lreal#0.2));
	{end}
END_TEST_S

TEST_S(test_xorshift32_nextGaussian)
	var
		prng: Xorshift32_t;
		ok: bool;
	end
	{st}
	// sigma < 0 → error
	Xorshift32_init(prng, dword#16#92d68ca2);
	Xorshift32_nextGaussian(self:=prng, mean:=lreal#0.0, sigma:=lreal#-1.0, eno=>ok);
	EXPECT_FALSE(ok);
	// sigma=0 → returns mean exactly
	Xorshift32_init(prng, dword#16#92d68ca2);
	EXPECT_NEAR(lreal#5.0, Xorshift32_nextGaussian(prng, lreal#5.0, lreal#0.0), lreal#1.0e-10);
	// Python reference value (seed=0x92d68ca2, mean=0, sigma=1)
	Xorshift32_init(prng, dword#16#92d68ca2);
	EXPECT_NEAR(lreal#-1.8733893786419715e+00, Xorshift32_nextGaussian(prng, lreal#0.0, lreal#1.0), lreal#1.0e-10);
	{end}
END_TEST_S
