TEST_S(test_linkedlist_dint_append)
	var
		result_eno: bool;
		xs_append: LinkedList_dint_t;
		xs_order: LinkedList_dint_t;
		xs_capacity: LinkedList_dint_t;
		idx: dint;
	end
	{st}

	// Fact: append adds elements up to capacity.
		// Arrange
	LinkedList_dint_init(xs_append);
	for idx := 1 to 8 by 1 do
		ASSERT_TRUE(LinkedList_dint_append(xs_append, idx));
	end_for;
		// Act / Assert
	EXPECT_EQ_INT32(8, LinkedList_dint_len(xs_append));

	// Fact: append preserves insertion order.
		// Arrange
	LinkedList_dint_init(xs_order);
	for idx := 1 to 8 by 1 do
		ASSERT_TRUE(LinkedList_dint_append(xs_order, idx));
	end_for;
	LinkedList_dint_get(self:=xs_order, index:=0, item:=idx, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Act / Assert
	EXPECT_EQ_INT32(1, idx);
	LinkedList_dint_get(self:=xs_order, index:=7, item:=idx, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(8, idx);

	// Fact: append fails at capacity limit.
		// Arrange
	LinkedList_dint_init(xs_capacity);
	for idx := 1 to 8 by 1 do
		ASSERT_TRUE(LinkedList_dint_append(xs_capacity, idx));
	end_for;
		// Act / Assert
	EXPECT_FALSE(LinkedList_dint_append(xs_capacity, 9));
	{end}
END_TEST_S

TEST_S(test_linkedlist_dint_pop)
	var
		result_eno: bool;
		xs_order: LinkedList_dint_t;
		xs_count: LinkedList_dint_t;
		xs_reuse: LinkedList_dint_t;
		item: dint;
	end
	{st}

	// Fact: pop removes and returns tail element.
		// Arrange
	LinkedList_dint_init(xs_order);
	ASSERT_TRUE(LinkedList_dint_append(xs_order, 1));
	ASSERT_TRUE(LinkedList_dint_append(xs_order, 2));
	ASSERT_TRUE(LinkedList_dint_append(xs_order, 3));
	ASSERT_TRUE(LinkedList_dint_pop(xs_order, item));
		// Act / Assert
	EXPECT_EQ_INT32(3, item);

	// Fact: pop decrements len and fails on empty list.
		// Arrange
	LinkedList_dint_init(xs_count);
	ASSERT_TRUE(LinkedList_dint_append(xs_count, 1));
	ASSERT_TRUE(LinkedList_dint_append(xs_count, 2));
	ASSERT_TRUE(LinkedList_dint_append(xs_count, 3));
	ASSERT_TRUE(LinkedList_dint_pop(xs_count, item));
		// Act / Assert
	EXPECT_EQ_INT32(2, LinkedList_dint_len(xs_count));
	ASSERT_TRUE(LinkedList_dint_pop(xs_count, item));
	EXPECT_EQ_INT32(2, item);
	ASSERT_TRUE(LinkedList_dint_pop(xs_count, item));
	EXPECT_EQ_INT32(1, item);
	EXPECT_FALSE(LinkedList_dint_pop(xs_count, item));
	EXPECT_TRUE(LinkedList_dint_empty(xs_count));

	// Fact: pop works correctly on single-element list.
		// Arrange
	LinkedList_dint_init(xs_reuse);
	ASSERT_TRUE(LinkedList_dint_append(xs_reuse, 9));
	ASSERT_TRUE(LinkedList_dint_pop(xs_reuse, item));
		// Act / Assert
	EXPECT_EQ_INT32(9, item);
	{end}
END_TEST_S

TEST_S(test_linkedlist_dint_insert)
	var
		result_eno: bool;
		xs_insert_ok: LinkedList_dint_t;
		xs_insert_range: LinkedList_dint_t;
		xs_insert_capacity: LinkedList_dint_t;
		idx: dint;
		value: dint;
	end
	{st}

	// Fact: insert adds element at head/middle/tail.
		// Arrange
	LinkedList_dint_init(xs_insert_ok);
	ASSERT_TRUE(LinkedList_dint_append(xs_insert_ok, 10));
	ASSERT_TRUE(LinkedList_dint_append(xs_insert_ok, 30));
	ASSERT_TRUE(LinkedList_dint_insert(xs_insert_ok, 1, 20));
	ASSERT_TRUE(LinkedList_dint_insert(xs_insert_ok, 0, 5));
	ASSERT_TRUE(LinkedList_dint_insert(xs_insert_ok, LinkedList_dint_len(xs_insert_ok), 40));
	LinkedList_dint_get(self:=xs_insert_ok, index:=0, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Act / Assert
	EXPECT_EQ_INT32(5, value);
	LinkedList_dint_get(self:=xs_insert_ok, index:=2, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(20, value);
	LinkedList_dint_get(self:=xs_insert_ok, index:=4, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(40, value);

	// Fact: insert fails on invalid index.
		// Arrange
	LinkedList_dint_init(xs_insert_range);
	ASSERT_TRUE(LinkedList_dint_append(xs_insert_range, 10));
	ASSERT_TRUE(LinkedList_dint_append(xs_insert_range, 30));
		// Act / Assert
	EXPECT_FALSE(LinkedList_dint_insert(xs_insert_range, -1, 99));
	EXPECT_FALSE(LinkedList_dint_insert(xs_insert_range, LinkedList_dint_len(xs_insert_range) + 1, 99));

	// Fact: insert fails at capacity limit.
		// Arrange
	LinkedList_dint_init(xs_insert_capacity);
	for idx := 1 to 8 by 1 do
		ASSERT_TRUE(LinkedList_dint_append(xs_insert_capacity, idx));
	end_for;
		// Act / Assert
	EXPECT_FALSE(LinkedList_dint_insert(xs_insert_capacity, 0, 99));
	EXPECT_FALSE(LinkedList_dint_insert(xs_insert_capacity, 4, 99));
	EXPECT_FALSE(LinkedList_dint_insert(xs_insert_capacity, LinkedList_dint_len(xs_insert_capacity), 99));
	EXPECT_EQ_INT32(8, LinkedList_dint_len(xs_insert_capacity));
	{end}
END_TEST_S

TEST_S(test_linkedlist_dint_hash)
	var
		result_eno: bool;
		empty_a: LinkedList_dint_t;
		empty_b: LinkedList_dint_t;
		same_a: LinkedList_dint_t;
		same_b: LinkedList_dint_t;
		diff_a: LinkedList_dint_t;
		diff_b: LinkedList_dint_t;
		hash_a: udint;
		hash_b: udint;
	end
	{st}

	// Fact: empty lists produce identical hash.
		// Arrange
	LinkedList_dint_init(empty_a);
	LinkedList_dint_init(empty_b);
	hash_a := LinkedList_dint_hash(empty_a);
	hash_b := LinkedList_dint_hash(empty_b);
		// Act / Assert
	EXPECT_EQ_UINT32(hash_a, hash_b);

	// Fact: hash values match for equivalent lists.
		// Arrange
	LinkedList_dint_init(same_a);
	LinkedList_dint_init(same_b);
	ASSERT_TRUE(LinkedList_dint_append(same_a, 10));
	ASSERT_TRUE(LinkedList_dint_append(same_a, 20));
	ASSERT_TRUE(LinkedList_dint_append(same_b, 10));
	ASSERT_TRUE(LinkedList_dint_append(same_b, 20));
	hash_a := LinkedList_dint_hash(same_a);
	hash_b := LinkedList_dint_hash(same_b);
		// Act / Assert
	EXPECT_EQ_UINT32(hash_a, hash_b);

	// Fact: hash differs for lists with different elements.
		// Arrange
	LinkedList_dint_init(diff_a);
	LinkedList_dint_init(diff_b);
	ASSERT_TRUE(LinkedList_dint_append(diff_a, 10));
	ASSERT_TRUE(LinkedList_dint_append(diff_a, 20));
	ASSERT_TRUE(LinkedList_dint_append(diff_b, 10));
	ASSERT_TRUE(LinkedList_dint_append(diff_b, 20));
	ASSERT_TRUE(LinkedList_dint_append(diff_b, 30));
	hash_a := LinkedList_dint_hash(diff_a);
	hash_b := LinkedList_dint_hash(diff_b);
		// Act / Assert
	EXPECT_NE_UINT32(hash_a, hash_b);
	{end}
END_TEST_S

TEST_S(test_linkedlist_string_caseinsensitive_usecase)
	var
		result_eno: bool;
		users: LinkedList_string_ci_t;
		users_same: LinkedList_string_ci_t;
		users_diff: LinkedList_string_ci_t;
		removed: string[16 + 1];
		item: string[16 + 1];
		hash_a: udint;
		hash_b: udint;
	end
	{st}

	// Fact: contains/index work case-insensitively.
		// Arrange
	LinkedList_string_ci_init(users);
	ASSERT_TRUE(LinkedList_string_ci_append(users, 'Alice'));
	ASSERT_TRUE(LinkedList_string_ci_append(users, 'Bob'));
		// Act / Assert
	EXPECT_TRUE(LinkedList_string_ci_contains(users, 'alice'));
	EXPECT_EQ_INT32(0, LinkedList_string_ci_index(users, 'ALICE'));

	// Fact: remove/get preserve original case.
		// Arrange
	ASSERT_TRUE(LinkedList_string_ci_remove(users, 0, removed));
		// Act / Assert
	EXPECT_TRUE(String_equals(removed, 'Alice'));
	EXPECT_EQ_INT32(1, LinkedList_string_ci_len(users));
	LinkedList_string_ci_get(self:=users, index:=0, item:=item, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_TRUE(String_equals(item, 'Bob'));

	// Fact: equals/hash work case-insensitively.
		// Arrange
	LinkedList_string_ci_init(users_same);
	LinkedList_string_ci_init(users_diff);
	ASSERT_TRUE(LinkedList_string_ci_append(users_same, 'bOb'));
	ASSERT_TRUE(LinkedList_string_ci_append(users_diff, 'Carol'));
		// Act / Assert
	EXPECT_TRUE(LinkedList_string_ci_equals(users, users_same));
	EXPECT_FALSE(LinkedList_string_ci_equals(users, users_diff));
	hash_a := LinkedList_string_ci_hash(users);
	hash_b := LinkedList_string_ci_hash(users_same);
	EXPECT_EQ_UINT32(hash_a, hash_b);
	{end}
END_TEST_S
