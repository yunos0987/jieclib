TEST_S(test_orderedmap_dint_upsert)
	var
		result_eno: bool;
		om_basic: OrderedMap_dint_t;
		om_empty_key: OrderedMap_dint_t;
		om_case_sensitive: OrderedMap_dint_t;
		om_capacity: OrderedMap_dint_t;
		replaced: bool;
		value: dint;
	end
	{st}

	// Fact: upsert handles insert, update, case-sensitivity, and capacity limit.
		// Arrange
	OrderedMap_dint_init(om_basic);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_basic, 'a', 10, replaced));
		// Assert
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_basic, 'a', 20, replaced));
	EXPECT_TRUE(replaced);
	OrderedMap_dint_init(om_empty_key);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_empty_key, '', -2, replaced));
	EXPECT_FALSE(replaced);
	OrderedMap_dint_lookup(self:=om_empty_key, key:='', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(-2, value);
	OrderedMap_dint_init(om_case_sensitive);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_case_sensitive, 'X', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_case_sensitive, 'x', 2, replaced));
	OrderedMap_dint_lookup(self:=om_case_sensitive, key:='X', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(1, value);
	OrderedMap_dint_lookup(self:=om_case_sensitive, key:='x', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(2, value);
	OrderedMap_dint_init(om_capacity);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_capacity, 'k1', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_capacity, 'k2', 2, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_capacity, 'k3', 3, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_capacity, 'k4', 4, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_capacity, 'k5', 5, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_capacity, 'k6', 6, replaced));
		// Act / Assert
	EXPECT_FALSE(OrderedMap_dint_upsert(om_capacity, 'k7', 7, replaced));
		// Assert
	EXPECT_EQ_INT32(6, OrderedMap_dint_len(om_capacity));
	{end}
END_TEST_S

TEST_S(test_orderedmap_dint_lookup)
	var
		result_eno: bool;
		om_hit: OrderedMap_dint_t;
		om_update: OrderedMap_dint_t;
		om_miss: OrderedMap_dint_t;
		replaced: bool;
		value: dint;
	end
	{st}

	// Fact: lookup returns value for registered key.
		// Arrange
	OrderedMap_dint_init(om_hit);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_hit, 'a', 10, replaced));
	OrderedMap_dint_lookup(self:=om_hit, key:='a', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(10, value);

	// Fact: lookup returns updated value after upsert.
		// Arrange
	OrderedMap_dint_init(om_update);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_update, 'a', 10, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_update, 'a', 77, replaced));
	ASSERT_TRUE(replaced);
	OrderedMap_dint_lookup(self:=om_update, key:='a', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(77, value);

	// Fact: lookup returns eno=false for unregistered key.
		// Arrange
	OrderedMap_dint_init(om_miss);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_miss, 'a', 10, replaced));
	OrderedMap_dint_lookup(self:=om_miss, key:='z', value:=value, eno=>result_eno);
		// Assert
	EXPECT_FALSE(result_eno);
	{end}
END_TEST_S

TEST_S(test_orderedmap_dint_delete)
	var
		result_eno: bool;
		om_hit: OrderedMap_dint_t;
		om_miss: OrderedMap_dint_t;
		om_head: OrderedMap_dint_t;
		keys: array[0..7] of string[16 + 1];
		written: dint;
		replaced: bool;
	end
	{st}

	// Fact: delete removes key and preserves order.
		// Arrange
	OrderedMap_dint_init(om_hit);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_hit, 'k1', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_hit, 'k2', 2, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_hit, 'k3', 3, replaced));
	ASSERT_TRUE(OrderedMap_dint_delete(om_hit, 'k2'));
		// Assert
	EXPECT_EQ_INT32(2, OrderedMap_dint_len(om_hit));
		// Act
	ASSERT_TRUE(OrderedMap_dint_keys(om_hit, keys, written));
		// Assert
	EXPECT_EQ_INT32(2, written);
	EXPECT_EQ_STRING('k1', keys[0]);
	EXPECT_EQ_STRING('k3', keys[1]);

	// Fact: delete returns false for unregistered key.
		// Arrange
	OrderedMap_dint_init(om_miss);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_miss, 'a', 1, replaced));
		// Act / Assert
	EXPECT_FALSE(OrderedMap_dint_delete(om_miss, 'z'));

	// Fact: delete updates order correctly when head is removed.
		// Arrange
	OrderedMap_dint_init(om_head);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_head, 'h1', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_head, 'h2', 2, replaced));
	ASSERT_TRUE(OrderedMap_dint_delete(om_head, 'h1'));
	ASSERT_TRUE(OrderedMap_dint_keys(om_head, keys, written));
		// Assert
	EXPECT_EQ_INT32(1, written);
	EXPECT_EQ_STRING('h2', keys[0]);
	{end}
END_TEST_S

TEST_S(test_orderedmap_dint_copy)
	var
		result_eno: bool;
		om_src1: OrderedMap_dint_t;
		om_dst1: OrderedMap_dint_t;
		om_src2: OrderedMap_dint_t;
		om_dst2: OrderedMap_dint_t;
		replaced: bool;
		keys: array[0..7] of string[16 + 1];
		written: dint;
		value: dint;
		om_src_empty: OrderedMap_dint_t;
		om_dst_empty: OrderedMap_dint_t;
	end
	{st}

	// Fact: copy duplicates all key-value pairs.
		// Arrange
	OrderedMap_dint_init(om_src1);
	OrderedMap_dint_init(om_dst1);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_src1, 'a', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_src1, 'b', 2, replaced));
	ASSERT_TRUE(OrderedMap_dint_copy(om_src1, om_dst1));
	ASSERT_TRUE(OrderedMap_dint_keys(om_dst1, keys, written));
		// Assert
	EXPECT_EQ_INT32(2, written);
	EXPECT_EQ_STRING('a', keys[0]);
	EXPECT_EQ_STRING('b', keys[1]);

	// Fact: copy creates independent instance.
		// Arrange
	OrderedMap_dint_init(om_src2);
	OrderedMap_dint_init(om_dst2);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_src2, 'a', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_copy(om_src2, om_dst2));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_src2, 'a', 9, replaced));
	OrderedMap_dint_lookup(self:=om_dst2, key:='a', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(1, value);

	// Fact: copy from empty map clears destination.
		// Arrange
	OrderedMap_dint_init(om_src_empty);
	OrderedMap_dint_init(om_dst_empty);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_dst_empty, 'z', 9, replaced));
	ASSERT_TRUE(OrderedMap_dint_copy(om_src_empty, om_dst_empty));
		// Assert
	EXPECT_TRUE(OrderedMap_dint_empty(om_dst_empty));
	{end}
END_TEST_S

		om_value2: OrderedMap_dint_t;
		replaced: bool;
		hash_a: udint;
		hash_b: udint;
	end
	{st}

	OrderedMap_dint_init(om_diff1);
	OrderedMap_dint_init(om_diff2);

	ASSERT_TRUE(OrderedMap_dint_upsert(om_diff1, 'b', 2, replaced));

	ASSERT_TRUE(OrderedMap_dint_upsert(om_diff2, 'a', 1, replaced));

	hash_b := OrderedMap_dint_hash(om_diff2);
		// Assert
	EXPECT_NE_UDINT(hash_a, hash_b);

	// Fact: hash values match for equivalent maps.
		// Arrange
	OrderedMap_dint_init(om_same1);
	OrderedMap_dint_init(om_same2);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_same1, 'a', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_same1, 'b', 2, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_same2, 'a', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_same2, 'b', 2, replaced));
		// Act
	hash_a := OrderedMap_dint_hash(om_same1);
	hash_b := OrderedMap_dint_hash(om_same2);
		// Assert
	EXPECT_EQ_UDINT(hash_a, hash_b);

	// Fact: hash differs for different values.
		// Arrange
	OrderedMap_dint_init(om_value1);
	OrderedMap_dint_init(om_value2);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_value1, 'a', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_value2, 'a', 9, replaced));
		// Act
	hash_a := OrderedMap_dint_hash(om_value1);
	hash_b := OrderedMap_dint_hash(om_value2);
		// Assert
	EXPECT_NE_UDINT(hash_a, hash_b);
	{end}
END_TEST_S

		replaced: bool;
	end
	{st}

	// Fact: empty is true after init.
		// Arrange
	OrderedMap_dint_init(om_init);
		// Assert
	EXPECT_TRUE(OrderedMap_dint_empty(om_init));

	// Fact: empty is false after upsert.
		// Arrange
	OrderedMap_dint_init(om_after_upsert);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_after_upsert, 'a', 1, replaced));
		// Assert
	EXPECT_FALSE(OrderedMap_dint_empty(om_after_upsert));

	// Fact: empty is true after clear.
		// Arrange
	OrderedMap_dint_init(om_after_clear);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_after_clear, 'a', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_clear(om_after_clear));
		// Assert
	EXPECT_TRUE(OrderedMap_dint_empty(om_after_clear));
	{end}
END_TEST_S

TEST_S(test_orderedmap_dint_clear)
	var
		result_eno: bool;
		om_clear: OrderedMap_dint_t;
		om_idempotent: OrderedMap_dint_t;
		om_reuse: OrderedMap_dint_t;
		replaced: bool;
	end
	{st}

	// Fact: clear sets len to 0.
		// Arrange
	OrderedMap_dint_init(om_clear);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_clear, 'a', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_clear, 'b', 2, replaced));
	ASSERT_TRUE(OrderedMap_dint_clear(om_clear));
		// Assert
	EXPECT_EQ_INT32(0, OrderedMap_dint_len(om_clear));

	// Fact: clear is idempotent.
		// Arrange
	OrderedMap_dint_init(om_idempotent);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_idempotent, 'a', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_clear(om_idempotent));
	ASSERT_TRUE(OrderedMap_dint_clear(om_idempotent));
		// Assert
	EXPECT_TRUE(OrderedMap_dint_empty(om_idempotent));

	// Fact: map is reusable after clear.
		// Arrange
	OrderedMap_dint_init(om_reuse);
	ASSERT_TRUE(OrderedMap_dint_upsert(om_reuse, 'a', 1, replaced));
	ASSERT_TRUE(OrderedMap_dint_clear(om_reuse));
	ASSERT_TRUE(OrderedMap_dint_upsert(om_reuse, 'b', 2, replaced));
		// Assert
	EXPECT_EQ_INT32(1, OrderedMap_dint_len(om_reuse));
	{end}
END_TEST_S

TEST_S(test_orderedmap_string_caseinsensitive_usecase)
	var
		result_eno: bool;
		users: OrderedMap_string_ci_t;
		replaced: bool;
		value: dint;
		keys: array[0..7] of string[16 + 1];
		written: dint;
	end
	{st}

	// Fact: case-insensitive upsert treats different cases as same key.
		// Arrange
	OrderedMap_string_ci_init(users);
	ASSERT_TRUE(OrderedMap_string_ci_upsert(users, 'Alice', 10, replaced));
		// Assert
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(OrderedMap_string_ci_upsert(users, 'aLiCe', 99, replaced));
	EXPECT_TRUE(replaced);
	EXPECT_EQ_INT32(1, OrderedMap_string_ci_len(users));
	OrderedMap_string_ci_lookup(self:=users, key:='ALICE', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(99, value);

	// Fact: contains/delete work case-insensitively.
		// Assert
	EXPECT_TRUE(OrderedMap_string_ci_contains(users, 'alice'));
		// Act / Assert
	EXPECT_TRUE(OrderedMap_string_ci_delete(users, 'AlIcE'));
		// Assert
	EXPECT_FALSE(OrderedMap_string_ci_contains(users, 'ALICE'));
	EXPECT_EQ_INT32(0, OrderedMap_string_ci_len(users));

	// Fact: keys preserve original case and insertion order.
		// Arrange
	ASSERT_TRUE(OrderedMap_string_ci_upsert(users, 'Bob', 1, replaced));
	ASSERT_TRUE(OrderedMap_string_ci_upsert(users, 'carol', 2, replaced));
	ASSERT_TRUE(OrderedMap_string_ci_upsert(users, 'DAVE', 3, replaced));
	ASSERT_TRUE(OrderedMap_string_ci_keys(users, keys, written));
		// Assert
	EXPECT_EQ_INT32(3, written);
	EXPECT_EQ_STRING('Bob', keys[0]);
	EXPECT_EQ_STRING('carol', keys[1]);
	EXPECT_EQ_STRING('DAVE', keys[2]);
	{end}
END_TEST_S
