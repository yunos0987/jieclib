TEST_S(test_arraylist_dint_append)
	var
		xs_append: ArrayList_dint_t;
		xs_order: ArrayList_dint_t;
		xs_capacity: ArrayList_dint_t;
		idx: dint;
		value: dint;
		result_eno: bool;
	end
	{st}

	// Fact: append succeeds and increments length up to capacity.
		// Arrange
	ArrayList_dint_init(xs_append);
		// Act
	for idx := 1 to 8 by 1 do
		ASSERT_TRUE(ArrayList_dint_append(xs_append, idx));
	end_for;
		// Assert
	EXPECT_EQ_INT32(8, ArrayList_dint_len(xs_append));

	// Fact: append preserves element order.
		// Arrange
	ArrayList_dint_init(xs_order);
		// Act
	ASSERT_TRUE(ArrayList_dint_append(xs_order, 10));
	ASSERT_TRUE(ArrayList_dint_append(xs_order, 20));
		// Act
	ArrayList_dint_get(self:=xs_order, index:=0, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(10, value);
		// Act
	ArrayList_dint_get(self:=xs_order, index:=1, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(20, value);

	// Fact: append fails when capacity is reached.
		// Arrange
	ArrayList_dint_init(xs_capacity);
	for idx := 1 to 8 by 1 do
		ASSERT_TRUE(ArrayList_dint_append(xs_capacity, idx));
	end_for;
		// Act / Assert
	EXPECT_FALSE(ArrayList_dint_append(xs_capacity, 9));
	{end}
END_TEST_S

TEST_S(test_arraylist_dint_insert)
	var
		xs_insert: ArrayList_dint_t;
		xs_oob: ArrayList_dint_t;
		xs_capacity: ArrayList_dint_t;
		value: dint;
		result_eno: bool;
		idx: dint;
	end
	{st}

	// Fact: insert succeeds at head, middle, and tail positions.
		// Arrange
	ArrayList_dint_init(xs_insert);
	ASSERT_TRUE(ArrayList_dint_append(xs_insert, 10));
	ASSERT_TRUE(ArrayList_dint_append(xs_insert, 30));
		// Act
	ASSERT_TRUE(ArrayList_dint_insert(xs_insert, 1, 20));
	ASSERT_TRUE(ArrayList_dint_insert(xs_insert, 0, 5));
	ASSERT_TRUE(ArrayList_dint_insert(xs_insert, ArrayList_dint_len(xs_insert), 40));
		// Assert
	ArrayList_dint_get(self:=xs_insert, index:=0, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(5, value);
	ArrayList_dint_get(self:=xs_insert, index:=2, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(20, value);
	ArrayList_dint_get(self:=xs_insert, index:=4, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(40, value);

	// Fact: insert fails for out-of-bounds index.
		// Arrange
	ArrayList_dint_init(xs_oob);
	ASSERT_TRUE(ArrayList_dint_append(xs_oob, 10));
		// Act / Assert
	EXPECT_FALSE(ArrayList_dint_insert(xs_oob, -1, 99));
	EXPECT_FALSE(ArrayList_dint_insert(xs_oob, ArrayList_dint_len(xs_oob) + 1, 99));

	// Fact: insert fails when capacity is full.
		// Arrange
	ArrayList_dint_init(xs_capacity);
	for idx := 1 to 8 by 1 do
		ASSERT_TRUE(ArrayList_dint_append(xs_capacity, idx));
	end_for;
		// Act / Assert
	EXPECT_FALSE(ArrayList_dint_insert(xs_capacity, 0, 99));
	EXPECT_FALSE(ArrayList_dint_insert(xs_capacity, ArrayList_dint_len(xs_capacity), 99));
	{end}
END_TEST_S

TEST_S(test_arraylist_dint_remove)
	var
		xs_remove: ArrayList_dint_t;

		xs_oob: ArrayList_dint_t;
		removed: dint;
		value: dint;
		result_eno: bool;
	end
	{st}

	// Fact: remove deletes element and shifts remaining elements.
		// Arrange
	ArrayList_dint_init(xs_remove);
	ASSERT_TRUE(ArrayList_dint_append(xs_remove, 10));
	ASSERT_TRUE(ArrayList_dint_append(xs_remove, 20));
	ASSERT_TRUE(ArrayList_dint_append(xs_remove, 30));
	ASSERT_TRUE(ArrayList_dint_append(xs_remove, 40));
		// Act
	ASSERT_TRUE(ArrayList_dint_remove(xs_remove, 1, removed));
		// Assert
	EXPECT_EQ_INT32(20, removed);
		// Act
	ASSERT_TRUE(ArrayList_dint_remove(xs_remove, 0, removed));
		// Assert
	EXPECT_EQ_INT32(10, removed);
		// Act
	ASSERT_TRUE(ArrayList_dint_remove(xs_remove, ArrayList_dint_len(xs_remove) - 1, removed));
		// Assert
	EXPECT_EQ_INT32(40, removed);
		// Act
	ArrayList_dint_get(self:=xs_remove, index:=0, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(30, value);

	// Fact: remove decrements length by 1.
		// Arrange
	ArrayList_dint_init(xs_count);
	ASSERT_TRUE(ArrayList_dint_append(xs_count, 10));
	ASSERT_TRUE(ArrayList_dint_append(xs_count, 20));
		// Act
	ASSERT_TRUE(ArrayList_dint_remove(xs_count, 0, removed));
		// Assert
	EXPECT_EQ_INT32(1, ArrayList_dint_len(xs_count));

	// Fact: remove fails for out-of-bounds index.
		// Arrange
	ArrayList_dint_init(xs_oob);
	ASSERT_TRUE(ArrayList_dint_append(xs_oob, 10));
		// Act / Assert
	EXPECT_FALSE(ArrayList_dint_remove(xs_oob, 1, removed));
	{end}
END_TEST_S

		values: array[0..7] of dint;
		small_values: array[0..1] of dint;
		written: dint;
		result_eno: bool;
	end
	{st}

	// Fact: values copies elements in order to output array.
		// Arrange
	ArrayList_dint_init(xs_order);

	ASSERT_TRUE(result_eno);
		// Assert

	EXPECT_EQ_INT32(10, values[0]);
	EXPECT_EQ_INT32(20, values[1]);
	EXPECT_EQ_INT32(30, values[2]);

	// Fact: values on empty list writes 0 and succeeds.
		// Arrange
	ArrayList_dint_init(xs_empty);
		// Act
	ASSERT_TRUE(ArrayList_dint_values(self:=xs_empty, values:=values, written=>written, eno=>result_eno));
	ASSERT_TRUE(result_eno);

		// Arrange
	ArrayList_dint_init(xs_small);
	ASSERT_TRUE(ArrayList_dint_append(xs_small, 1));
	ASSERT_TRUE(ArrayList_dint_append(xs_small, 2));
	ASSERT_TRUE(ArrayList_dint_append(xs_small, 3));
		// Act / Assert
	EXPECT_FALSE(ArrayList_dint_values(self:=xs_small, values:=small_values, written=>written, eno=>result_eno));

TEST_S(test_arraylist_dint_copy)
	var
		src_equal: ArrayList_dint_t;
		dst_equal: ArrayList_dint_t;
		src_indep: ArrayList_dint_t;
		dst_indep: ArrayList_dint_t;
		src_empty: ArrayList_dint_t;
		dst_empty: ArrayList_dint_t;
		value: dint;
		result_eno: bool;
	end
	{st}

	// Fact: copy duplicates all elements.
		// Arrange
	ArrayList_dint_init(src_equal);
	ArrayList_dint_init(dst_equal);
	ASSERT_TRUE(ArrayList_dint_append(src_equal, 7));
	ASSERT_TRUE(ArrayList_dint_append(src_equal, 8));
		// Act
	ASSERT_TRUE(ArrayList_dint_copy(src_equal, dst_equal));
	ArrayList_dint_get(self:=dst_equal, index:=0, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(7, value);

	// Fact: copy creates independent instance.
		// Arrange
	ArrayList_dint_init(src_indep);
	ArrayList_dint_init(dst_indep);
	ASSERT_TRUE(ArrayList_dint_append(src_indep, 7));
		// Act
	ASSERT_TRUE(ArrayList_dint_copy(src_indep, dst_indep));
	ASSERT_TRUE(ArrayList_dint_set(src_indep, 0, 70));
		// Act
	ArrayList_dint_get(self:=src_indep, index:=0, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(70, value);
		// Act
	ArrayList_dint_get(self:=dst_indep, index:=0, item:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(7, value);

	// Fact: copy from empty list clears destination.
		// Arrange
	ArrayList_dint_init(src_empty);
	ArrayList_dint_init(dst_empty);
	ASSERT_TRUE(ArrayList_dint_append(dst_empty, 1));
		// Act
	ASSERT_TRUE(ArrayList_dint_copy(src_empty, dst_empty));
		// Assert
	EXPECT_TRUE(ArrayList_dint_empty(dst_empty));
	{end}
END_TEST_S

	EXPECT_NE_UINT32(hash_a, hash_b);

TEST_S(test_arraylist_string_caseinsensitive_usecase)
	var
		users: ArrayList_string_ci_t;
		users_same: ArrayList_string_ci_t;
		users_diff: ArrayList_string_ci_t;
		removed: string[16 + 1];
		value: string[16 + 1];
		result_eno: bool;
		hash_a: udint;
		hash_b: udint;
	end
	{st}

	// Fact: contains and index are case-insensitive.
		// Arrange
	ArrayList_string_ci_init(users);

	ASSERT_TRUE(ArrayList_string_ci_append(users, 'Bob'));
