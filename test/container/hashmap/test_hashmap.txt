TEST_S(test_hashmap_dint_init)
	var
		result_eno: bool;
		map1: HashMap_dint_t;
		map2: HashMap_dint_t;
		map3: HashMap_dint_t;
		keys: array[0..7] of string[16 + 1];
		written: dint;
		value: dint;
		replaced: bool;
	end
	{st}

	// Fact: init sets len to 0 and contains returns false for any key.
		// Arrange
	HashMap_dint_init(map1);
		// Act / Assert
	EXPECT_EQ_INT32(0, HashMap_dint_len(map1));
	EXPECT_FALSE(HashMap_dint_contains(map1, ''));
	EXPECT_FALSE(HashMap_dint_contains(map1, 'key'));

	// Fact: re-init clears existing entries.
		// Arrange
	HashMap_dint_init(map2);
	ASSERT_TRUE(HashMap_dint_upsert(map2, 'key', 2, replaced));
		// Act / Assert
	EXPECT_TRUE(HashMap_dint_contains(map2, 'key'));
	HashMap_dint_init(map2); // re-init
	EXPECT_EQ_INT32(0, HashMap_dint_len(map2));
	EXPECT_FALSE(HashMap_dint_contains(map2, ''));
	EXPECT_FALSE(HashMap_dint_contains(map2, 'key'));

	// Fact: init results in keys returning empty array.
		// Arrange
	HashMap_dint_init(map3);
	ASSERT_TRUE(HashMap_dint_keys(map3, keys, written));
		// Act / Assert
	EXPECT_EQ_INT32(0, written);
	{end}
END_TEST_S

TEST_S(test_hashmap_dint_upsert)
	var
		result_eno: bool;
		map_basic: HashMap_dint_t;
		map_empty_key: HashMap_dint_t;
		map_update: HashMap_dint_t;
		map_case_sensitive: HashMap_dint_t;
		map_capacity: HashMap_dint_t;
		value: dint;
		replaced: bool;
	end
	{st}

	// Fact: upsert handles insert, update, case-sensitivity, and capacity limit.
		// Arrange
	HashMap_dint_init(map_basic);
	ASSERT_TRUE(HashMap_dint_upsert(map_basic, 'u1', 2, replaced));
		// Act / Assert
	EXPECT_FALSE(replaced);
	HashMap_dint_lookup(self:=map_basic, key:='u1', value:=value, eno=>result_eno);
	EXPECT_TRUE(result_eno);
	EXPECT_EQ_INT32(2, value);
	HashMap_dint_init(map_empty_key);
	ASSERT_TRUE(HashMap_dint_upsert(map_empty_key, '', -2, replaced));
	EXPECT_FALSE(replaced);
	HashMap_dint_lookup(self:=map_empty_key, key:='', value:=value, eno=>result_eno);
	EXPECT_TRUE(result_eno);
	EXPECT_EQ_INT32(-2, value);
	HashMap_dint_init(map_update);
	ASSERT_TRUE(HashMap_dint_upsert(map_update, 'u1', 2, replaced));
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(HashMap_dint_upsert(map_update, 'u1', 3, replaced));
	EXPECT_TRUE(replaced);
	HashMap_dint_lookup(self:=map_update, key:='u1', value:=value, eno=>result_eno);
	EXPECT_TRUE(result_eno);
	EXPECT_EQ_INT32(3, value);
	HashMap_dint_init(map_case_sensitive);
	ASSERT_TRUE(HashMap_dint_upsert(map_case_sensitive, 'X', 2, replaced));
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(HashMap_dint_upsert(map_case_sensitive, 'x', 3, replaced));
	EXPECT_FALSE(replaced);
	HashMap_dint_lookup(self:=map_case_sensitive, key:='X', value:=value, eno=>result_eno);
	EXPECT_TRUE(result_eno);
	EXPECT_EQ_INT32(2, value);
	HashMap_dint_lookup(self:=map_case_sensitive, key:='x', value:=value, eno=>result_eno);
	EXPECT_TRUE(result_eno);
	EXPECT_EQ_INT32(3, value);
	HashMap_dint_init(map_capacity);
	ASSERT_TRUE(HashMap_dint_upsert(map_capacity, 'c1', 1, replaced));
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(HashMap_dint_upsert(map_capacity, 'c2', 2, replaced));
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(HashMap_dint_upsert(map_capacity, 'c3', 3, replaced));
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(HashMap_dint_upsert(map_capacity, 'c4', 4, replaced));
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(HashMap_dint_upsert(map_capacity, 'c5', 5, replaced));
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(HashMap_dint_upsert(map_capacity, 'c6', 6, replaced));
	EXPECT_FALSE(replaced);
	EXPECT_EQ_INT32(6, HashMap_dint_len(map_capacity));
	EXPECT_FALSE(HashMap_dint_upsert(map_capacity, 'c7', 7, replaced));
	EXPECT_EQ_INT32(6, HashMap_dint_len(map_capacity));
	EXPECT_FALSE(HashMap_dint_contains(map_capacity, 'c7'));
	{end}
END_TEST_S

TEST_S(test_hashmap_dint_copy)
	var
		result_eno: bool;
		map_src: HashMap_dint_t;
		map_dst: HashMap_dint_t;
		map_src_empty: HashMap_dint_t;
		map_dst_empty: HashMap_dint_t;
		value: dint;
		replaced: bool;
	end
	{st}

	// Fact: copy duplicates all key-value pairs.
		// Arrange
	HashMap_dint_init(map_src);
	HashMap_dint_init(map_dst);
	ASSERT_TRUE(HashMap_dint_upsert(map_src, 'a', 1, replaced));
	ASSERT_TRUE(HashMap_dint_upsert(map_src, 'b', 2, replaced));
	ASSERT_TRUE(HashMap_dint_copy(map_src, map_dst));
		// Act / Assert
	EXPECT_EQ_INT32(2, HashMap_dint_len(map_dst));
	HashMap_dint_lookup(self:=map_dst, key:='a', value:=value, eno=>result_eno);
	EXPECT_TRUE(result_eno);
	EXPECT_EQ_INT32(1, value);
	HashMap_dint_lookup(self:=map_dst, key:='b', value:=value, eno=>result_eno);
	EXPECT_TRUE(result_eno);
	EXPECT_EQ_INT32(2, value);

	// Fact: copy creates independent instance.
		// Arrange
	ASSERT_TRUE(HashMap_dint_upsert(map_src, 'a', 99, replaced));
	HashMap_dint_lookup(self:=map_src, key:='a', value:=value, eno=>result_eno);
		// Act / Assert
	EXPECT_TRUE(result_eno);
	EXPECT_EQ_INT32(99, value);
	HashMap_dint_lookup(self:=map_dst, key:='a', value:=value, eno=>result_eno);
	EXPECT_TRUE(result_eno);
	EXPECT_EQ_INT32(1, value);

	// Fact: copy from empty map clears destination.
		// Arrange
	HashMap_dint_init(map_src_empty);
	HashMap_dint_init(map_dst_empty);
	ASSERT_TRUE(HashMap_dint_upsert(map_dst_empty, 'z', 9, replaced));
	ASSERT_TRUE(HashMap_dint_copy(map_src_empty, map_dst_empty));
		// Act / Assert
	EXPECT_EQ_INT32(0, HashMap_dint_len(map_dst_empty));
	{end}
END_TEST_S

TEST_S(test_hashmap_dint_delete)
	var
		result_eno: bool;
		map_empty: HashMap_dint_t;
		map_notempty: HashMap_dint_t;
		map_case_sensitive: HashMap_dint_t;
		replaced: bool;
	end
	{st}

	// Fact: delete removes existing keys and leaves state stable on misses.
		// Arrange
	HashMap_dint_init(map_empty);
		// Act / Assert
	EXPECT_FALSE(HashMap_dint_delete(map_empty, ''));
	EXPECT_EQ_INT32(0, HashMap_dint_len(map_empty));
	EXPECT_FALSE(HashMap_dint_delete(map_empty, 'x'));
	EXPECT_EQ_INT32(0, HashMap_dint_len(map_empty));
	HashMap_dint_init(map_notempty);
	ASSERT_TRUE(HashMap_dint_upsert(map_notempty, 'x', 2, replaced));
	ASSERT_TRUE(HashMap_dint_upsert(map_notempty, 'y', 3, replaced));
	EXPECT_EQ_INT32(2, HashMap_dint_len(map_notempty));
	ASSERT_TRUE(HashMap_dint_delete(map_notempty, 'x'));
	EXPECT_FALSE(HashMap_dint_contains(map_notempty, 'x'));
	EXPECT_TRUE(HashMap_dint_contains(map_notempty, 'y'));
	EXPECT_EQ_INT32(1, HashMap_dint_len(map_notempty));
	EXPECT_FALSE(HashMap_dint_delete(map_notempty, 'x'));
	EXPECT_EQ_INT32(1, HashMap_dint_len(map_notempty));
	HashMap_dint_init(map_case_sensitive);
	ASSERT_TRUE(HashMap_dint_upsert(map_case_sensitive, 'X', 2, replaced));
	EXPECT_FALSE(HashMap_dint_delete(map_case_sensitive, 'x'));
	EXPECT_EQ_INT32(1, HashMap_dint_len(map_case_sensitive));
	{end}
END_TEST_S

		map3: HashMap_dint_t;

TEST_S(test_hashmap_dint_keys)
	var
		result_eno: bool;
		map_empty: HashMap_dint_t;
		map_notempty: HashMap_dint_t;
		map_after_remove: HashMap_dint_t;
		keys: array[0..7] of string[16 + 1];
		written: dint;
		replaced: bool;
		idx: dint;
		found_k1: bool;
		found_k2: bool;
		found_k3: bool;
		found_k4: bool;
	end
	{st}

	// Fact: keys returns empty array for empty map.
		// Arrange
	HashMap_dint_init(map_empty);
		// Act / Assert
	EXPECT_TRUE(HashMap_dint_keys(map_empty, keys, written));
	EXPECT_EQ_INT32(0, written);

	// Fact: keys returns all registered keys.
		// Arrange
	HashMap_dint_init(map_notempty);
	ASSERT_TRUE(HashMap_dint_upsert(map_notempty, 'k1', 1, replaced));
	ASSERT_TRUE(HashMap_dint_upsert(map_notempty, 'k2', 2, replaced));
	ASSERT_TRUE(HashMap_dint_upsert(map_notempty, 'k3', 3, replaced));
	found_k1 := false;
	found_k2 := false;
	found_k3 := false;
		// Act / Assert
	EXPECT_TRUE(HashMap_dint_keys(map_notempty, keys, written));
	EXPECT_EQ_INT32(3, written);
	for idx := 0 to written - 1 by 1 do
		if keys[idx] = 'k1' then
			found_k1 := true;
		elsif keys[idx] = 'k2' then
			found_k2 := true;
		elsif keys[idx] = 'k3' then
			found_k3 := true;
		end_if;
	end_for;
	EXPECT_TRUE(found_k1);
	EXPECT_TRUE(found_k2);
	EXPECT_TRUE(found_k3);

	// Fact: keys updates correctly after delete.
		// Arrange
	HashMap_dint_init(map_after_remove);
	ASSERT_TRUE(HashMap_dint_upsert(map_after_remove, 'k1', 1, replaced));
	ASSERT_TRUE(HashMap_dint_upsert(map_after_remove, 'k4', 4, replaced));
	ASSERT_TRUE(HashMap_dint_delete(map_after_remove, 'k1'));
	found_k4 := false;
		// Act / Assert
	EXPECT_TRUE(HashMap_dint_keys(map_after_remove, keys, written));
	EXPECT_EQ_INT32(1, written);
	if written = 1 then
		if keys[0] = 'k4' then
			found_k4 := true;
		end_if;
	end_if;
	EXPECT_TRUE(found_k4);
	{end}
END_TEST_S

	var

		map3: HashMap_dint_t;
		map4: HashMap_dint_t;

		hash3: udint;

	// Fact: hash values match for equivalent maps.
		// Arrange
	HashMap_dint_init(map1);
	HashMap_dint_init(map2);

	HashMap_dint_init(map4);

TEST_S(test_hashmap_dint_empty)
	var
		result_eno: bool;
		map_empty: HashMap_dint_t;
		map_notempty: HashMap_dint_t;
		map_after_remove: HashMap_dint_t;
		replaced: bool;
	end
	{st}

	// Fact: empty is true immediately after init.
		// Arrange
	HashMap_dint_init(map_empty);
		// Act / Assert
	EXPECT_TRUE(HashMap_dint_empty(map_empty));

	// Fact: empty is false after upsert.
		// Arrange
	HashMap_dint_init(map_notempty);
	ASSERT_TRUE(HashMap_dint_upsert(map_notempty, 'k1', 1, replaced));
