TEST_S(test_linkedhashmap_dint_upsert)
	var
		result_eno: bool;
		map_insert_update: LinkedHashMap_dint_t;
		map_capacity: LinkedHashMap_dint_t;
		map_nohash_insert: LinkedHashMap_dint_nohash_t;
		map_nohash_update: LinkedHashMap_dint_nohash_t;
		map_nohash_capacity: LinkedHashMap_dint_nohash_t;
		replaced: bool;
	end
	{st}

	// Fact: upsert handles insert, update, and capacity edge cases.
		// Arrange
	LinkedHashMap_dint_init(map_insert_update);
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_insert_update, key:='a', value:=10, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_insert_update, key:='a', value:=20, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(result_eno);
	EXPECT_TRUE(replaced);
	EXPECT_EQ_INT32(1, map_insert_update.count);
	EXPECT_TRUE(String_equals(map_insert_update.nodes[map_insert_update.head_idx].key, 'a'));
	EXPECT_EQ_INT32(20, map_insert_update.nodes[map_insert_update.head_idx].value);
	LinkedHashMap_dint_init(map_capacity);
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_capacity, key:='k0', value:=0, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_capacity, key:='k1', value:=1, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_capacity, key:='k2', value:=2, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_capacity, key:='k3', value:=3, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_capacity, key:='k4', value:=4, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_capacity, key:='k5', value:=5, replaced=>replaced, eno=>result_eno));
		// Act / Assert
	EXPECT_FALSE(LinkedHashMap_dint_upsert(self:=map_capacity, key:='k6', value:=6, replaced=>replaced, eno=>result_eno));
		// Assert
	EXPECT_EQ_INT32(6, map_capacity.count);

	// Fact: nohash upsert inserts new key.
		// Arrange
	LinkedHashMap_dint_nohash_init(map_nohash_insert);
	ASSERT_TRUE(LinkedHashMap_dint_nohash_upsert(self:=map_nohash_insert, key:='z', value:=9, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_FALSE(replaced);
	EXPECT_EQ_INT32(1, map_nohash_insert.count);

	// Fact: nohash upsert updates existing key with replaced=true.
		// Arrange
	LinkedHashMap_dint_nohash_init(map_nohash_update);
	ASSERT_TRUE(LinkedHashMap_dint_nohash_upsert(self:=map_nohash_update, key:='u', value:=1, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_nohash_upsert(self:=map_nohash_update, key:='u', value:=2, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_TRUE(replaced);
	EXPECT_EQ_INT32(1, map_nohash_update.count);

	// Fact: nohash upsert fails at capacity limit.
		// Arrange
	LinkedHashMap_dint_nohash_init(map_nohash_capacity);
	ASSERT_TRUE(LinkedHashMap_dint_nohash_upsert(self:=map_nohash_capacity, key:='k0', value:=0, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_nohash_upsert(self:=map_nohash_capacity, key:='k1', value:=1, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_nohash_upsert(self:=map_nohash_capacity, key:='k2', value:=2, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_nohash_upsert(self:=map_nohash_capacity, key:='k3', value:=3, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_nohash_upsert(self:=map_nohash_capacity, key:='k4', value:=4, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_nohash_upsert(self:=map_nohash_capacity, key:='k5', value:=5, replaced=>replaced, eno=>result_eno));
		// Act / Assert
	EXPECT_FALSE(LinkedHashMap_dint_nohash_upsert(self:=map_nohash_capacity, key:='k6', value:=6, replaced=>replaced, eno=>result_eno));
		// Assert
	EXPECT_EQ_INT32(6, map_nohash_capacity.count);
	{end}
END_TEST_S

TEST_S(test_linkedhashmap_dint_lookup)
	var
		result_eno: bool;
		map_found: LinkedHashMap_dint_t;
		map_missing: LinkedHashMap_dint_t;
		map_updated: LinkedHashMap_dint_t;
		map_access_move: LinkedHashMap_dint_access_t;
		map_access_miss: LinkedHashMap_dint_access_t;
		map_access_single: LinkedHashMap_dint_access_t;
		replaced: bool;
		value: dint;
		keys: array[8] of string[16 + 1];
		written: dint;
	end
	{st}

	// Fact: lookup returns value for registered key.
		// Arrange
	LinkedHashMap_dint_init(map_found);
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_found, key:='a', value:=20, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(result_eno);
	LinkedHashMap_dint_lookup(self:=map_found, key:='a', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(20, value);

	// Fact: lookup returns eno=false for unregistered key.
		// Arrange
	LinkedHashMap_dint_init(map_missing);
	LinkedHashMap_dint_lookup(self:=map_missing, key:='missing', value:=value, eno=>result_eno);
		// Assert
	EXPECT_FALSE(result_eno);

	// Fact: lookup returns updated value after upsert.
		// Arrange
	LinkedHashMap_dint_init(map_updated);
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_updated, key:='u', value:=1, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_updated, key:='u', value:=9, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(replaced);
	LinkedHashMap_dint_lookup(self:=map_updated, key:='u', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(9, value);

	// Fact: access-order lookup moves accessed key to tail.
		// Arrange
	LinkedHashMap_dint_access_init(map_access_move);
	ASSERT_TRUE(LinkedHashMap_dint_access_upsert(self:=map_access_move, key:='a', value:=1, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_access_upsert(self:=map_access_move, key:='b', value:=2, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_access_upsert(self:=map_access_move, key:='c', value:=3, replaced=>replaced, eno=>result_eno));
	LinkedHashMap_dint_access_lookup(self:=map_access_move, key:='a', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	ASSERT_TRUE(LinkedHashMap_dint_access_keys(self:=map_access_move, keys:=keys, written=>written, eno=>result_eno));
		// Assert
	EXPECT_EQ_INT32(3, written);
	EXPECT_TRUE(String_equals(keys[0], 'b'));
	EXPECT_TRUE(String_equals(keys[1], 'c'));
	EXPECT_TRUE(String_equals(keys[2], 'a'));

	// Fact: access-order lookup returns eno=false for unregistered key.
		// Arrange
	LinkedHashMap_dint_access_init(map_access_miss);
	ASSERT_TRUE(LinkedHashMap_dint_access_upsert(self:=map_access_miss, key:='a', value:=1, replaced=>replaced, eno=>result_eno));
	LinkedHashMap_dint_access_lookup(self:=map_access_miss, key:='missing', value:=value, eno=>result_eno);
		// Assert
	EXPECT_FALSE(result_eno);

	// Fact: access-order lookup on single-element map keeps order.
		// Arrange
	LinkedHashMap_dint_access_init(map_access_single);
	ASSERT_TRUE(LinkedHashMap_dint_access_upsert(self:=map_access_single, key:='only', value:=7, replaced=>replaced, eno=>result_eno));
	LinkedHashMap_dint_access_lookup(self:=map_access_single, key:='only', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	ASSERT_TRUE(LinkedHashMap_dint_access_keys(self:=map_access_single, keys:=keys, written=>written, eno=>result_eno));
		// Assert
	EXPECT_EQ_INT32(1, written);
	EXPECT_TRUE(String_equals(keys[0], 'only'));
	{end}
END_TEST_S

		map_add_count: LinkedHashMap_dint_t;
		map_update_remove_count: LinkedHashMap_dint_t;
		replaced: bool;
	end
	{st}

	// Fact: len is 0 after init.
		// Arrange
	LinkedHashMap_dint_init(map_init_count);
		// Assert

	// Fact: len increments after upsert.
		// Arrange
	LinkedHashMap_dint_init(map_add_count);

		// Assert
	EXPECT_EQ_INT32(2, LinkedHashMap_dint_len(map_add_count));

	// Fact: len stays unchanged on update and decrements on delete.
		// Arrange
	LinkedHashMap_dint_init(map_update_remove_count);

		// Act
	ASSERT_TRUE(LinkedHashMap_dint_delete(self:=map_update_remove_count, key:='a', eno=>result_eno));

	{end}

TEST_S(test_linkedhashmap_dint_delete)
	var
		result_eno: bool;
		map_remove_hit: LinkedHashMap_dint_t;
		map_remove_miss: LinkedHashMap_dint_t;
		map_remove_head: LinkedHashMap_dint_t;
		replaced: bool;
		keys: array[8] of string[16 + 1];
		written: dint;
		removed: bool;
	end
	{st}

	// Fact: delete removes existing key and preserves order.
		// Arrange
	LinkedHashMap_dint_init(map_remove_hit);
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_remove_hit, key:='k1', value:=1, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_remove_hit, key:='k2', value:=2, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_remove_hit, key:='k3', value:=3, replaced=>replaced, eno=>result_eno));
	removed := LinkedHashMap_dint_delete(self:=map_remove_hit, key:='k2', eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_TRUE(removed);
	EXPECT_EQ_INT32(2, map_remove_hit.count);
		// Act
	ASSERT_TRUE(LinkedHashMap_dint_keys(self:=map_remove_hit, keys:=keys, written=>written, eno=>result_eno));
		// Assert
	EXPECT_EQ_INT32(2, written);
	EXPECT_TRUE(String_equals(keys[0], 'k1'));
	EXPECT_TRUE(String_equals(keys[1], 'k3'));

	// Fact: delete returns false for unregistered key.
		// Arrange
	LinkedHashMap_dint_init(map_remove_miss);
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_remove_miss, key:='k1', value:=1, replaced=>replaced, eno=>result_eno));
	removed := LinkedHashMap_dint_delete(self:=map_remove_miss, key:='missing', eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_FALSE(removed);
	EXPECT_EQ_INT32(1, map_remove_miss.count);

	// Fact: delete updates head correctly when head is removed.
		// Arrange
	LinkedHashMap_dint_init(map_remove_head);
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_remove_head, key:='h1', value:=1, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_remove_head, key:='h2', value:=2, replaced=>replaced, eno=>result_eno));

TEST_S(test_linkedhashmap_dint_equals)
	var
		result_eno: bool;

		map_c: LinkedHashMap_dint_t;
		map_d: LinkedHashMap_dint_t;
		replaced: bool;
	end
	{st}

	// Fact: equals returns true for same key-value pairs in same order.
		// Arrange
	LinkedHashMap_dint_init(map_a);
	LinkedHashMap_dint_init(map_b);
	LinkedHashMap_dint_init(map_c);
	LinkedHashMap_dint_init(map_d);
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_a, key:='k1', value:=1, replaced=>replaced, eno=>result_eno));

	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_b, key:='k1', value:=1, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_b, key:='k2', value:=2, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_c, key:='k2', value:=2, replaced=>replaced, eno=>result_eno));

TEST_S(test_linkedhashmap_dint_hash)
	var
		result_eno: bool;
		map_a: LinkedHashMap_dint_t;
		map_b: LinkedHashMap_dint_t;
		map_c: LinkedHashMap_dint_t;
		map_d: LinkedHashMap_dint_t;

		map_f: LinkedHashMap_dint_t;
		replaced: bool;
		hash_a: udint;
		hash_b: udint;
	end
	{st}

	// Fact: hash values match for equivalent maps.
		// Arrange
	LinkedHashMap_dint_init(map_a);
	LinkedHashMap_dint_init(map_b);
	ASSERT_TRUE(LinkedHashMap_dint_upsert(self:=map_a, key:='k1', value:=1, replaced=>replaced, eno=>result_eno));

TEST_S(test_linkedhashmap_string_caseinsensitive_usecase)
	var
		result_eno: bool;
		sessions: LinkedHashMap_string_ci_t;
		replaced: bool;
		value: dint;
		keys: array[8] of string[16 + 1];
		written: dint;
		removed: bool;
	end
	{st}

	// Fact: case-insensitive upsert treats different cases as same key.
		// Arrange
	LinkedHashMap_string_ci_init(sessions);
	ASSERT_TRUE(LinkedHashMap_string_ci_upsert(self:=sessions, key:='Alice', value:=1, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_FALSE(replaced);
	ASSERT_TRUE(LinkedHashMap_string_ci_upsert(self:=sessions, key:='aLiCe', value:=2, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(result_eno);
	EXPECT_TRUE(replaced);
	EXPECT_EQ_INT32(1, LinkedHashMap_string_ci_len(sessions));
	LinkedHashMap_string_ci_lookup(self:=sessions, key:='ALICE', value:=value, eno=>result_eno);
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(2, value);

	// Fact: contains/delete work case-insensitively.
		// Assert
	EXPECT_TRUE(LinkedHashMap_string_ci_contains(self:=sessions, key:='alice', eno=>result_eno));
	ASSERT_TRUE(result_eno);
		// Act
	removed := LinkedHashMap_string_ci_delete(self:=sessions, key:='AlIcE', eno=>result_eno);
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_TRUE(removed);
	EXPECT_FALSE(LinkedHashMap_string_ci_contains(self:=sessions, key:='ALICE', eno=>result_eno));
	ASSERT_TRUE(result_eno);
	EXPECT_EQ_INT32(0, LinkedHashMap_string_ci_len(sessions));

	// Fact: keys preserve original case and insertion order.
		// Arrange
	ASSERT_TRUE(LinkedHashMap_string_ci_upsert(self:=sessions, key:='Bob', value:=10, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_string_ci_upsert(self:=sessions, key:='Carol', value:=20, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_string_ci_upsert(self:=sessions, key:='dave', value:=30, replaced=>replaced, eno=>result_eno));
	ASSERT_TRUE(LinkedHashMap_string_ci_keys(self:=sessions, keys:=keys, written=>written, eno=>result_eno));
	ASSERT_TRUE(result_eno);
		// Assert
	EXPECT_EQ_INT32(3, written);
	EXPECT_TRUE(String_equals(keys[0], 'Bob'));
	EXPECT_TRUE(String_equals(keys[1], 'Carol'));
	EXPECT_TRUE(String_equals(keys[2], 'dave'));
	{end}
END_TEST_S
